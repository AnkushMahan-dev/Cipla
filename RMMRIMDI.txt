DEFINE messages_context_data_set.

  CALL METHOD ref_process_log->context_data_get

  gs_context_data-document = gs_log_variables-document.

DEFINE message_with_context_data.
  IF ref_process_log IS INITIAL.
    CREATE OBJECT ref_process_log.
  ENDIF.

  IF NOT &6 EQ space.
    gs_message-level = &6.
  ENDIF.

  gs_log_variables-document = &1.
  gs_log_variables-item = &2.
  CONDENSE gs_log_variables-item.
  gs_log_variables-subitem = &3.
  CONDENSE gs_log_variables-subitem.
  CONCATENATE gs_log_variables-document
              gs_log_variables-item
              gs_log_variables-subitem
         INTO gs_log_variables-docvariant.
  CONDENSE gs_log_variables-docvariant.

  CALL METHOD ref_process_log->context_data_get
****** Rel 60E_SP7 - ECIP #21432
*    exporting
    IMPORTING
******
      es_context_data = gs_context_data.

  gs_context_data-document = gs_log_variables-document.
  gs_context_data-docvariant = gs_log_variables-docvariant.
  gs_context_data-context-tabname = &4.
  gs_context_data-context-value = &5.



  CALL METHOD ref_process_log->message_add
    EXPORTING
      i_level                 = gs_message-level
      i_msgid                 = sy-msgid
      i_msgty                 = sy-msgty
      i_msgno                 = sy-msgno
      i_msgv1                 = sy-msgv1
      i_msgv2                 = sy-msgv2
      i_msgv3                 = sy-msgv3
      i_msgv4                 = sy-msgv4
      i_default_context_data  = space
      is_context_data         = gs_context_data
    EXCEPTIONS
      message_to_be_displayed = 1
      OTHERS                  = 2.
  IF sy-subrc EQ 1
  AND NOT sy-msgid IS INITIAL
  AND NOT sy-msgty IS INITIAL
  AND NOT sy-msgno IS INITIAL.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
               WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  ENDIF.
END-OF-DEFINITION.

DEFINE message_simple.
  IF ref_process_log IS INITIAL.
    CREATE OBJECT ref_process_log.
  ENDIF.

  IF NOT &1 EQ space.
    gs_message-level = &1.
  ENDIF.

  CALL METHOD ref_process_log->message_add
    EXPORTING
      i_level                 = gs_message-level
      i_msgid                 = sy-msgid
      i_msgty                 = sy-msgty
      i_msgno                 = sy-msgno
      i_msgv1                 = sy-msgv1
      i_msgv2                 = sy-msgv2
      i_msgv3                 = sy-msgv3
      i_msgv4                 = sy-msgv4
    EXCEPTIONS
      message_to_be_displayed = 1
      OTHERS                  = 2.
  IF sy-subrc EQ 1
  AND NOT sy-msgid IS INITIAL
  AND NOT sy-msgty IS INITIAL.
*  and not sy-msgno is initial.
****Rel E SP1
    IF sy-msgid EQ '/IRM/GMSG1' OR
       sy-msgid EQ '/IRM/GMSG2'.
      CLEAR gs_log_variables-text.
      CALL FUNCTION '/IRM/GMSG_TEXTGET'
        EXPORTING
          i_msgcls             = sy-msgid
          i_msgno              = sy-msgno
          i_placeholder1       =  sy-msgv1
          i_placeholder2       =  sy-msgv2
          i_placeholder3       =  sy-msgv3
          i_placeholder4       =  sy-msgv4
       IMPORTING
         e_message           = gs_log_variables-text.

      MESSAGE gs_log_variables-text TYPE sy-msgty.
    ELSE.
***
      MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
                 WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
*** Rel E SP1
    ENDIF.
***
  ENDIF.

END-OF-DEFINITION.
