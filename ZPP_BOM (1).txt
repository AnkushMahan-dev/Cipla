FORM get_data .

  SELECT a~belnr
         a~gjahr
         a~bukrs
         a~blart
         a~budat
         a~bldat
         a~cpudt
*         a~augdt
         a~xblnr
         b~kunnr
         b~val_frm
         b~val_to
    FROM bsid AS a
    INNER JOIN zfi_intc_ven_cus AS b
    ON b~kunnr EQ a~kunnr
    INTO CORRESPONDING FIELDS OF TABLE ta_bsid
    WHERE budat IN s_budat
    AND   bukrs IN s_bukrs.


  SELECT a~belnr
         a~gjahr
         a~bukrs
         a~blart
         a~budat
         a~bldat
         a~cpudt
*         a~augdt
         a~xblnr
         b~kunnr
         b~val_frm
         b~val_to
    FROM bsad AS a
    INNER JOIN zfi_intc_ven_cus AS b
    ON b~kunnr EQ a~kunnr
    APPENDING CORRESPONDING FIELDS OF TABLE ta_bsid
    WHERE budat IN s_budat
    AND   bukrs IN s_bukrs.


  SELECT a~belnr
         a~gjahr
         a~bukrs
         a~blart
         a~budat
         a~bldat
         a~cpudt
*         a~augdt
         a~xblnr
         b~lifnr
         b~val_frm
         b~val_to
    FROM bsik AS a
    INNER JOIN zfi_intc_ven_cus AS b
    ON b~lifnr EQ a~lifnr
    APPENDING CORRESPONDING FIELDS OF TABLE ta_bsid
    WHERE budat IN s_budat
    AND   bukrs IN s_bukrs.

  SELECT a~belnr
         a~gjahr
         a~bukrs
         a~blart
         a~budat
         a~bldat
         a~cpudt
*         a~xblnr
*         a~augdt
         a~xblnr
         b~lifnr
         b~val_frm
         b~val_to
    FROM bsak AS a
    INNER JOIN zfi_intc_ven_cus AS b
    ON b~lifnr EQ a~lifnr
    APPENDING CORRESPONDING FIELDS OF TABLE ta_bsid
    WHERE budat IN s_budat
    AND   bukrs IN s_bukrs.


  IF ta_bsid IS NOT INITIAL.

    SORT ta_bsid BY belnr gjahr bukrs.

    SELECT bukrs
           belnr
           gjahr
           buzei
           buzid
           augdt
           augbl
           bschl
           koart
           umskz
           shkzg
           dmbtr
           wrbtr
           pswsl
           zuonr
           sgtxt
           vbund
           hkont
           kunnr
           lifnr
           ebeln
           prctr
      FROM bseg AS a
      INTO CORRESPONDING FIELDS OF TABLE ta_bseg
      FOR ALL ENTRIES IN ta_bsid
      WHERE belnr EQ ta_bsid-belnr
      AND   bukrs EQ ta_bsid-bukrs
      AND   gjahr EQ ta_bsid-gjahr.


    IF ta_bseg IS NOT INITIAL.

      SORT ta_bseg BY belnr bukrs gjahr.

      SELECT kunnr name1
        FROM kna1
        INTO CORRESPONDING FIELDS OF TABLE ta_kna1
        FOR ALL ENTRIES IN ta_bseg
        WHERE kunnr EQ ta_bseg-kunnr.
      IF sy-subrc EQ 0.
        SORT ta_kna1 BY kunnr.
      ENDIF.

      SELECT lifnr name1
        FROM lfa1
        INTO CORRESPONDING FIELDS OF TABLE ta_lfa1
        FOR ALL ENTRIES IN ta_bseg
        WHERE lifnr EQ ta_bseg-lifnr.
      IF sy-subrc EQ 0.
        SORT ta_lfa1 BY lifnr.
      ENDIF.

      SELECT gl_frm gl_to tag tag_rpt
        FROM zfi_intc_gl
        INTO CORRESPONDING FIELDS OF TABLE ta_zfi_intc_gl
        FOR ALL ENTRIES IN ta_bseg
        WHERE ( ( gl_frm LE ta_bseg-hkont
        OR       gl_frm GE ta_bseg-hkont )
        AND     ( gl_to  LE ta_bseg-hkont
        OR      gl_to GE ta_bseg-hkont ) ).

      IF sy-subrc EQ 0.
        SORT ta_zfi_intc_gl BY gl_frm.
        DELETE ta_zfi_intc_gl WHERE gl_frm IS INITIAL AND gl_to IS INITIAL.
        ta_zfi_intc_gl1[] = ta_zfi_intc_gl[].
      ENDIF.

    ENDIF.

    SELECT blart tag
      FROM zfi_intc_dt
      INTO CORRESPONDING FIELDS OF TABLE ta_zfi_intc_dt
      FOR ALL ENTRIES IN ta_bsid
      WHERE blart EQ ta_bsid-blart.

    IF sy-subrc EQ 0.
      SORT ta_zfi_intc_dt BY blart.
    ENDIF.

    SELECT saknr txt50 FROM skat
      INTO TABLE ta_skat
      FOR ALL ENTRIES IN ta_bseg
      WHERE saknr EQ ta_bseg-hkont.

    IF sy-subrc EQ 0.
      SORT ta_skat BY saknr.
    ENDIF.

    SELECT bukrs belnr gjahr waers FROM bkpf
      INTO TABLE ta_bkpf
      FOR ALL ENTRIES IN ta_bseg
      WHERE bukrs EQ ta_bseg-bukrs
      AND   belnr EQ ta_bseg-belnr
      AND   gjahr EQ ta_bseg-gjahr.

    IF sy-subrc EQ 0.
      SORT ta_bkpf BY belnr bukrs gjahr.
    ENDIF.


  ENDIF.
ENDFORM.

FORM process_elimination_data.

  DATA(ta_data1) = ta_data[].
  REFRESH:ta_data.

  IF p_eli = 'X'.
    r_eli-sign = 'I'.
    r_eli-option = 'EQ'.
    r_eli-low = 'YES'.
    APPEND r_eli.

  ELSE.
    r_eli-sign = 'I'.
    r_eli-option = 'EQ'.
    r_eli-low = 'YES'.
    APPEND r_eli.
    r_eli-sign = 'I'.
    r_eli-option = 'EQ'.
    r_eli-low = 'NO'.
    APPEND r_eli.
  ENDIF.

  SELECT * FROM zfi_intc_ven_cus
    INTO TABLE @DATA(ta_ven_cus)
    FOR ALL ENTRIES IN @ta_data1
    WHERE kunnr = @ta_data1-kunnr AND
          lifnr = @ta_data1-lifnr AND
          elimination IN @r_eli .

  IF ta_ven_cus IS NOT INITIAL.

    SELECT * FROM zfi_intc_categry
      INTO TABLE @DATA(ta_categry)
      FOR ALL ENTRIES IN @ta_ven_cus
      WHERE category = @ta_ven_cus-category.
    IF sy-subrc = 0.
      SORT ta_categry BY category.
    ENDIF.
  ENDIF.

  SORT ta_data1 BY kunnr lifnr.
  SORT ta_ven_cus BY kunnr lifnr.
  SORT ta_categry BY category.

  CLEAR:wa_data.
  LOOP AT ta_data1 INTO DATA(wa_data1).
    READ TABLE ta_ven_cus INTO DATA(wa_ven_cus1) WITH KEY kunnr = wa_data1-kunnr
                                                          lifnr = wa_data1-lifnr
                                                           BINARY SEARCH.
    IF sy-subrc = 0.
      wa_data1-elimination =  wa_ven_cus1-elimination.
      wa_data1-tradingp =  wa_ven_cus1-vbund.
      READ TABLE ta_categry INTO DATA(wa_categry) WITH KEY category = wa_ven_cus1-category.
      IF sy-subrc = 0.
        wa_data1-categorydesc = wa_categry-categorydesc.
      ENDIF.
      MOVE-CORRESPONDING wa_data1 TO wa_data.
      APPEND wa_data TO ta_data.
    ENDIF.
    CLEAR:wa_data,wa_data1.
  ENDLOOP.

ENDFORM.
