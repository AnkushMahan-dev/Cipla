*----------------------------------------------------------------------*
*       CLASS lcx_main_exception DEFINITION
*----------------------------------------------------------------------*
*
*----------------------------------------------------------------------*
CLASS lcx_main_exception DEFINITION
  INHERITING FROM cx_static_check.

  PUBLIC SECTION.

    INTERFACES lif_text .

    ALIASES get_full_text FOR lif_text~get_full_text.

    TYPES ty_text TYPE c LENGTH 200.
    TYPES ty_text_variable TYPE c LENGTH 50.

    DATA text TYPE ty_text READ-ONLY .
    DATA text_v1 TYPE ty_text_variable READ-ONLY .
    DATA text_v2 TYPE ty_text_variable READ-ONLY .
    DATA text_v3 TYPE ty_text_variable READ-ONLY .
    DATA text_v4 TYPE ty_text_variable READ-ONLY .

    METHODS constructor
      IMPORTING
        !textid   LIKE textid OPTIONAL
        !previous LIKE previous OPTIONAL
        !text     TYPE ty_text OPTIONAL
        !text_v1  TYPE ty_text_variable OPTIONAL
        !text_v2  TYPE ty_text_variable OPTIONAL
        !text_v3  TYPE ty_text_variable OPTIONAL
        !text_v4  TYPE ty_text_variable OPTIONAL.

    METHODS if_message~get_text REDEFINITION.
ENDCLASS.                    "lcx_main_exception DEFINITION

*----------------------------------------------------------------------*
*       CLASS lcx_main_exception IMPLEMENTATION
*----------------------------------------------------------------------*
*
*----------------------------------------------------------------------*
CLASS lcx_main_exception IMPLEMENTATION.
  METHOD constructor.
    CALL METHOD super->constructor
      EXPORTING
        textid   = textid
        previous = previous.

    me->text = text.
    me->text_v1 = text_v1.
    me->text_v2 = text_v2.
    me->text_v3 = text_v3.
    me->text_v4 = text_v4.
  ENDMETHOD.                    "constructor

  METHOD lif_text~get_full_text.
    rv_text = text.

    REPLACE '&1' IN rv_text WITH text_v1.
    REPLACE '&2' IN rv_text WITH text_v2.
    REPLACE '&3' IN rv_text WITH text_v3.
    REPLACE '&4' IN rv_text WITH text_v4.
  ENDMETHOD.                    "lif_text~get_full_text

  METHOD if_message~get_text.
    IF text IS NOT INITIAL.
      result = get_full_text( ).
    ELSE.
      result = super->get_text( ).
    ENDIF.
  ENDMETHOD.                    "if_message~get_text
ENDCLASS.                    "lcx_main_exception IMPLEMENTATION

  METHOD check_variant.
    DATA: lv_xml_content     TYPE xstring,
          lt_conv_targ_stack TYPE ty_conv_target_stack_tab,
          ls_target_stack    TYPE ty_conv_target_stack_str,
          l_name             TYPE string,

          l_lines            TYPE i,
          lt_inst_prod_ver   TYPE tt_instswprod,
          ls_prod            LIKE LINE OF lt_inst_prod_ver.

    CONSTANTS: lc_simdb_rel TYPE string VALUE 'SYCM_REL_STACK'.


" messages shouldn't be longer than 128 char, otherwise they are truncated

* Check current S4 ATC variant
    FREE               lt_inst_prod_ver.
    "Get installed Product Versions
    CALL FUNCTION 'OCS_GET_INSTALLED_SWPRODUCTS'
      IMPORTING
        et_swproducts = lt_inst_prod_ver
      EXCEPTIONS
        OTHERS        = 0.
    DESCRIBE TABLE lt_inst_prod_ver LINES l_lines.
    CHECK l_lines NE 0.

    " search specifically for S4CORE
    CLEAR l_var.
    LOOP AT lt_inst_prod_ver INTO ls_prod.
      IF ls_prod-name CS 'S4HANA ON PREMISE' OR ls_prod-name CS 'SAP S/4HANA PRIVATE CLOUD'.
        CONCATENATE 'S4HANA_READINESS_' ls_prod-version INTO l_var. "#EC NOTEXT
      ENDIF.
    ENDLOOP.

    "think positive!
    rv_ok = 'X'.

* check if that variant exists here
    SELECT SINGLE *
      FROM scichkv_hd
      INTO ls_variant
      WHERE checkvname LIKE l_var.
    IF sy-subrc NE 0.
* variant does not exist, abort.
      APPEND 'There is no check variant for the ABAP test cockpit available for the source release. See SAP Note 2436688 for more information.' TO _messages. "#EC NOTEXT
      rv_ok = ' '.
    ENDIF.

* Get S/4HANA conversion target release
    CALL FUNCTION 'FUNCTION_EXISTS'
      EXPORTING
        funcname           = if_tmw_rc_collect_bpa=>c_fm-sitem_targ_stk_get
      EXCEPTIONS
        function_not_exist = 1
        OTHERS             = 2.
    IF sy-subrc <> 0.
      APPEND 'The function module /SDF/GEN_FUNCS_S4_TARG_STK_GET is not available in this system. Implement SAP Note 2399707.' TO _messages.
      rv_ok = ' '.
      RETURN. "we cannot proceed from here as the FM is required for the next checks
    ENDIF.

    "Get S/4HANA conversion target release
    CALL FUNCTION if_tmw_rc_collect_bpa=>c_fm-sitem_targ_stk_get
      IMPORTING
        ev_result_xstr = lv_xml_content.
    IF lv_xml_content IS NOT INITIAL.
      CALL TRANSFORMATION id
        SOURCE XML lv_xml_content
        RESULT conv_targ_stack = lt_conv_targ_stack.
    ENDIF.
    " get the entry corresponding to the input param of the report
    READ TABLE lt_conv_targ_stack INTO ls_target_stack
    WITH KEY stack_number = p_prd_st.
    CHECK sy-subrc EQ 0.
    " try to get the corresponding S/4HANA version from the name
    CLEAR l_var.
    l_len = strlen( ls_target_stack-prod_ver_name ) - 4.
    l_name = ls_target_stack-prod_ver_name+l_len(4).
    CONCATENATE 'S4HANA_READINESS_' l_name '%' INTO l_var.

* check if that variant exists here
    SELECT SINGLE *
      FROM scichkv_hd
      INTO ls_variant
      WHERE checkvname LIKE l_var.
    IF sy-subrc NE 0.
* variant does not exist, abort.
      APPEND 'There is no check variant for the ABAP test cockpit available for the target release. See SAP Note 2436688 for more information.' TO _messages. "#EC NOTEXT
      rv_ok = ' '.
    ENDIF.

* check if simDB is up to date
    SELECT COUNT(*) FROM (LC_SIMDB_REL)
      INTO l_lines
      WHERE release_id LIKE l_var.
    IF sy-subrc ne 0.
      APPEND 'The simplification database content is not up to date in this system. For more information, see SAP Note 2241080.' TO _messages. "#EC NOTEXT
      rv_ok = ' '.
    ENDIF.

  METHOD update_wp_parameter.
    DATA:
      lv_mode                    TYPE boolean,
      lv_total_nbr_wp            TYPE int4,
      lv_total_nbr_wp_string     TYPE string,
      lv_nbr_available_wp        TYPE int4,
      lv_nbr_available_wp_string TYPE string,
      lv_date                    TYPE sy-datum,
      lv_date_string             TYPE c LENGTH 20,
      lv_time                    TYPE tims,
      lv_time_string             TYPE c LENGTH 20,
      lv_nbr_job_figl            TYPE int4,

    IF gv_fdq_prerequisites_fulfilled <> abap_true.
      RETURN.

    IF p_fdqpa = 'X'.
      lv_mode = abap_true.
    ELSE.
      lv_mode = abap_false.
    ENDIF.

    c1fdqji = lcl_fdq_data_coll=>cs_wp_text-text_1.
    c2fdqji = lcl_fdq_data_coll=>cs_wp_text-text_2.
    c3fdqji = lcl_fdq_data_coll=>cs_wp_text-text_3.

    CALL METHOD ('CL_RC_FDQ_HANDLER')=>('GET_JOB_PARAM_PER_MODE')
      EXPORTING
        iv_fiaa             = p_fdqaa
        iv_figl             = p_fdqgl
        is_para_mode        = lv_mode
      IMPORTING
        ev_nbr_available_wp = lv_nbr_available_wp
        ev_total_nbr_wp     = lv_total_nbr_wp
        ev_date             = lv_date
        ev_time             = lv_time
        ev_nbr_job_figl     = lv_nbr_job_figl
        ev_nbr_job_fiaa     = lv_nbr_job_fiaa
        ev_nbr_job_fiml     = lv_nbr_job_fiml
      EXCEPTIONS
        work_process_error  = 1
        OTHERS              = 99.

    IF sy-subrc <> 0.
      p_fdqglj = 1.
      p_fdqaaj = 1.
      p_fdqmlj = 1.

      MESSAGE 'Failure to calculate work process' TYPE 'W'.

      RETURN.
    ENDIF.

    p_fdqglj = lv_nbr_job_figl.
    p_fdqaaj = lv_nbr_job_fiaa.
    p_fdqmlj = lv_nbr_job_fiml.

    "Update global variables for WP check
    gv_fdq_nbr_available_wp = lv_nbr_available_wp.
    gv_fdq_total_nbr_wp     = lv_total_nbr_wp.
    gv_fdq_date_wp_check    = lv_date.
    gv_fdq_time_wp_check    = lv_time.


    WRITE lv_date TO lv_date_string.
    REPLACE 'DD.MM.YYYY' IN c2fdqji WITH lv_date_string.

    WRITE lv_time TO lv_time_string.
    REPLACE 'hh.mm.ss' IN c2fdqji WITH lv_time_string.

    lv_total_nbr_wp_string     = lv_total_nbr_wp.
    REPLACE 'Y' IN c1fdqji WITH lv_total_nbr_wp_string.

    lv_nbr_available_wp_string = lv_nbr_available_wp.
    REPLACE 'X' IN c1fdqji WITH lv_nbr_available_wp_string.
    REPLACE 'X' IN c3fdqji WITH lv_nbr_available_wp_string.

    CONDENSE c1fdqji.
    CONDENSE c2fdqji.
    CONDENSE c3fdqji.
  ENDMETHOD.                    "update_wp_parameter

      lv_expiration_age_in_second    TYPE i,
      lv_message                     TYPE string,
      lv_sting                       TYPE string.

    CLEAR ev_buffered_data.

    IF iv_client_dependent = abap_true.
      ls_data = cl_rc_collect_client_handler=>select_w_clt_considration(
        iv_trigid     = iv_trigid

        iv_subid      = iv_subid

        FROM srtm_datax
        INTO ls_data
       WHERE trigid     = iv_trigid
         AND trigoffset = iv_trigoffset
         AND subid      = iv_subid.

      IF sy-subrc <> 0.
        RETURN.
      ENDIF.
    ENDIF.

    IF ls_data IS INITIAL OR xstrlen( ls_data-xtext ) = 0.
      RETURN.
    ENDIF.

    GET TIME STAMP FIELD lv_current_timestamp.

    CONVERT DATE ls_data-ddate TIME ls_data-dtime
      INTO TIME STAMP lv_old_timestamp TIME ZONE sy-zonlo.

    CALL FUNCTION 'TIMECALC_DIFF'
      EXPORTING
        timestamp1 = lv_old_timestamp
        timestamp2 = lv_current_timestamp
        timezone   = sy-zonlo
      IMPORTING
        difference = lv_buffered_data_age_in_second.

      add_info_mesg lv_message.

      ev_data_in_buffer_is_old = abap_false.

    ENDIF.

      WITH p_idoc_s  = p_idoc_s

      WITH p_ifwebs  = p_ifwebs
      WITH p_if_odt = lv_is_odata_selected "p_if_odt

      WITH p_if_min  = p_if_min
    WITH p_tr_usg  = p_tr_usg
    WITH p_h_size  = p_h_size

*----------------------------------------------------------------------*
*      -->PV_REPORT_NAME  text
*----------------------------------------------------------------------*
FORM collect_bp_data USING pv_report_name TYPE progname.
  DATA:
    lv_xml           TYPE string,
    lv_xsl           TYPE string,
    lv_xml_xstring   TYPE xstring,
    lv_xsl_xstring   TYPE xstring,
    lv_buffered      TYPE flag VALUE abap_false,
    lv_timestamp_cur TYPE timestamp,
    lv_timestamp_old TYPE timestamp,
    lv_str_tmp       TYPE string,
    lv_age_second    TYPE i,
    lv_exp_second    TYPE i.

  "Collect the data
  add_info_mesg 'Start to collect Business Partner data'.

  CLEAR ls_data.
  ls_data = cl_rc_collect_client_handler=>select_w_clt_considration(
      iv_trigid     = c_data_key-data_trigid
      iv_trigoffset = c_data_key-data_trigoffset
      iv_subid      = c_data_key-subid_bpnxml_data
  ).

  IF ls_data IS NOT INITIAL AND xstrlen( ls_data-xtext ) > 0.
    GET TIME STAMP FIELD lv_timestamp_cur.

    CONVERT DATE ls_data-ddate TIME ls_data-dtime
      INTO TIME STAMP lv_timestamp_old TIME ZONE sy-zonlo.

    CALL FUNCTION 'TIMECALC_DIFF'
      EXPORTING
        timestamp1 = lv_timestamp_old
        timestamp2 = lv_timestamp_cur
        timezone   = sy-zonlo
      IMPORTING
        difference = lv_age_second.
    lv_exp_second = 60 * 60 * 24 * gv_expired_dat.

    IF lv_age_second <= lv_exp_second.
      lv_mesg_text = 'Buffered data less than 30 days( &1 ) is used.'.
      lv_str_tmp = lv_timestamp_old.
      REPLACE '&1' IN lv_mesg_text WITH lv_str_tmp.
      MESSAGE lv_mesg_text TYPE 'I'.

      lv_xml_xstring = ls_data-xtext.
      CLEAR ls_data.

      ls_data = cl_rc_collect_client_handler=>select_w_clt_considration(
          iv_trigid     = c_data_key-data_trigid
          iv_trigoffset = c_data_key-data_trigoffset
          iv_subid      = c_data_key-subid_bpnxsl_data
      ).

      lv_xsl_xstring = ls_data-xtext.
      lv_buffered = abap_true.
    ENDIF.
  ENDIF.

  IF lv_buffered <> abap_true.
    CLEAR lv_mesg_text.
    CALL FUNCTION 'CVI_READINESS_CHECK'                     "#EC EXISTS
      IMPORTING
        ev_xml               = lv_xml
        ev_xsl               = lv_xsl
      EXCEPTIONS
        transformation_error = 1
        OTHERS               = 2.
    IF sy-subrc <> 0.
      MESSAGE 'Business Partner call function CVI_READINESS_CHECK failed' TYPE 'E' DISPLAY LIKE 'I'.
      RETURN.
    ELSE.
      CLEAR ls_data.
      CALL FUNCTION 'SCMS_STRING_TO_XSTRING'
        EXPORTING
          text   = lv_xml
        IMPORTING
          buffer = lv_xml_xstring
        EXCEPTIONS
          OTHERS = 0.

      TRY.
          cl_rc_collect_client_handler=>modify_w_clt_considration(
              iv_trigid     = c_data_key-data_trigid
              iv_trigoffset = c_data_key-data_trigoffset
              iv_subid      = c_data_key-subid_bpnxml_data
              iv_ddate      = sy-datum
              iv_dtime      = sy-uzeit
              iv_xtext      = lv_xml_xstring
          ).
        CATCH cx_rc_collect_client_handler.
      ENDTRY.

      CLEAR ls_data.
      CALL FUNCTION 'SCMS_STRING_TO_XSTRING'
        EXPORTING
          text   = lv_xsl
        IMPORTING
          buffer = lv_xsl_xstring
        EXCEPTIONS
          OTHERS = 0.

      TRY.
          cl_rc_collect_client_handler=>modify_w_clt_considration(
            EXPORTING
              iv_trigid     = c_data_key-data_trigid
              iv_trigoffset = c_data_key-data_trigoffset
              iv_subid      = c_data_key-data_trigoffset
              iv_ddate      = sy-datum
              iv_dtime      = sy-uzeit
              iv_xtext      = lv_xsl_xstring
          ).
        CATCH cx_rc_collect_client_handler.
      ENDTRY.

      "Delete old BP data once new bp collects with success
      SELECT SINGLE *
        FROM srtm_datax
        INTO ls_data
      WHERE trigid     = c_data_key-data_trigid
        AND trigoffset = c_data_key-data_trigoffset
        AND subid      = c_data_key-subid_bpxml_data.

      IF sy-subrc = 0 AND xstrlen( ls_data-xtext ) > 0.
        DELETE FROM srtm_datax
          WHERE trigid     = c_data_key-data_trigid
            AND trigoffset = c_data_key-data_trigoffset
            AND subid      = c_data_key-subid_bpxml_data.
      ENDIF.

      SELECT SINGLE *
        FROM srtm_datax
        INTO ls_data
      WHERE trigid     = c_data_key-data_trigid
        AND trigoffset = c_data_key-data_trigoffset
        AND subid      = c_data_key-subid_bpxsl_data.

      IF sy-subrc = 0 AND xstrlen( ls_data-xtext ) > 0.
        DELETE FROM srtm_datax
          WHERE trigid     = c_data_key-data_trigid
            AND trigoffset = c_data_key-data_trigoffset
            AND subid      = c_data_key-subid_bpxsl_data.
      ENDIF.
    ENDIF.
  ENDIF.

  "Add xml data to the ZIP file
  cl_tmw_rc_collect_bpa=>add_content_to_zip(
    iv_file_name   = 'bpccn.xml'
    iv_content     = lv_xml_xstring
    iv_coll_report = 'RC_COLLECT_ANALYSIS_DATA' ).
  "Add xsl data to the ZIP file
  "bpcc.xsl related with bpccn.xml as design
  cl_tmw_rc_collect_bpa=>add_content_to_zip(
    iv_file_name   = 'bpcc.xsl'
    iv_content     = lv_xsl_xstring
    iv_coll_report = 'RC_COLLECT_ANALYSIS_DATA' ).

  add_info_mesg 'Business Partner data is collected'.
ENDFORM.                    "collect_bp_data

*&      Form  schedule_sub_job
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->PV_JOB_NAME  text
*      -->PV_JOB_TYPE  text
*----------------------------------------------------------------------*
FORM schedule_sub_job  USING pv_job_name      TYPE btcjob
                             pv_job_type      TYPE char10.

  DATA:
    lv_job_count     TYPE btcjobcnt,
    ls_sched_sub_job LIKE LINE OF cl_tmw_rc_collect_bpa=>st_sched_sub_job.

*--------------------------------------------------------------------*
* Schedule data collector in batch mode

  "Open job-planning without dialog
  CALL FUNCTION 'SET_PRINT_PARAMETERS'
    EXPORTING
      immediately = ''.

  CALL FUNCTION 'JOB_OPEN'
    EXPORTING
      jobname  = pv_job_name
    IMPORTING
      jobcount = lv_job_count
    EXCEPTIONS
      OTHERS   = 1.
  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
      WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
    RETURN.
  ENDIF.

  SUBMIT (sy-repid) "rc_collect_analysis_data
    WITH p_subjob = pv_job_type
    WITH p_prd_st = p_prd_st
    WITH so_bpa_k  IN so_bpa_k
    USER sy-uname
    VIA JOB pv_job_name NUMBER lv_job_count
    AND RETURN.

  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
      WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
    RETURN.
  ENDIF.

  "Close the job
  CALL FUNCTION 'JOB_CLOSE'
    EXPORTING
      jobcount  = lv_job_count
      jobname   = pv_job_name
      strtimmed = 'X'
    EXCEPTIONS
      OTHERS    = 1.
  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
      WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
    RETURN.
  ENDIF.

  CLEAR ls_sched_sub_job.
  ls_sched_sub_job-count = lv_job_count.
  ls_sched_sub_job-name  = pv_job_name.
  ls_sched_sub_job-type  = pv_job_type.

  CASE pv_job_type.
    WHEN c_sub_job_type-sic_relv OR c_sub_job_type-sic_cons.
      ls_sched_sub_job-descr = 'Simplification Item Check'.
    WHEN c_sub_job_type-hana_size.
      ls_sched_sub_job-descr = 'HANA Sizing'.
    WHEN c_sub_job_type-bpa.
      ls_sched_sub_job-descr = 'Business Process Analytics'.
    WHEN c_sub_job_type-dvm.
      ls_sched_sub_job-descr = 'Data Volume Management'.
    WHEN c_sub_job_type-innova.
      ls_sched_sub_job-descr = 'Innovation Potential'.
    WHEN c_sub_job_type-fdq.
      ls_sched_sub_job-descr = 'Financial Data Quality'.
    WHEN c_sub_job_type-sic_efd.
      ls_sched_sub_job-descr = 'Simplification Item Effort Driver'.
    WHEN OTHERS.
  ENDCASE.

  APPEND ls_sched_sub_job TO cl_tmw_rc_collect_bpa=>st_sched_sub_job.

ENDFORM.                    "schedule_sub_job

*&      Form  validate_input
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      <--CV_STOP_PROCESS  text
*----------------------------------------------------------------------*
FORM validate_input CHANGING cv_stop_process TYPE boolean.
  "Currently only for FDQ
  DATA lv_scope_sel_mode TYPE ty_scope_sel_mode.

  "check WP input
  DATA lv_total_parallel_jobs_nbr TYPE int4 VALUE 0.

  IF p_fdq <> abap_true.
    EXIT.
  ENDIF.

  lv_scope_sel_mode = go_scope_sel_mode->get_scope_sel_mode( ).


  IF lv_scope_sel_mode <> lcl_scope_sel_mode=>cs_scope_sel_mode-standard.
    IF p_fdqgl = 'X'.
      IF p_fdqglj < 1.
        MESSAGE 'Wrong number of jobs for General Ledger' TYPE 'W'.

        EXIT.
      ENDIF.

      IF p_fdqgle < 1.
        MESSAGE 'Wrong number of entities for General Ledger' TYPE 'W'.

        EXIT.
      ENDIF.
    ENDIF.

    IF p_fdqaa = 'X'.
      IF p_fdqaaj  < 1 OR p_fdqaae  < 1.
        MESSAGE 'Wrong number of jobs for Asset Accounting' TYPE 'W'.

        EXIT.
      ENDIF.

      IF p_fdqaae  < 1.
        MESSAGE 'Wrong number of entities for Asset Accounting' TYPE 'W'.

        EXIT.
      ENDIF.
    ENDIF.

    IF p_fdqml = 'X'.
      IF p_fdqmlj  < 1 OR p_fdqmle  < 1.
        MESSAGE 'Wrong number of jobs for Material Ledger' TYPE 'W'.

        EXIT.
      ENDIF.

      IF p_fdqmle  < 1.
        MESSAGE 'Wrong number of entities for Material Ledger' TYPE 'W'.

        EXIT.
      ENDIF.
    ENDIF.

    lv_total_parallel_jobs_nbr = lcl_fdq_data_coll=>get_total_parallel_jobs_nbr( ).

    IF lv_total_parallel_jobs_nbr > gv_fdq_nbr_available_wp.
      PERFORM show_popup_fdq_wp USING    lv_total_parallel_jobs_nbr
                                CHANGING cv_stop_process.
    ENDIF.
  ENDIF.
ENDFORM.                    "validate_input

FORM collect_usage_data .    "8/10

  CONSTANTS: subid_dpt     TYPE int4 VALUE 3100.            "#EC NOTEXT

  DATA:
    lv_data_in_buffer_is_old TYPE boolean VALUE abap_true,
    lo_dpt_api               TYPE REF TO lcl_dpt_data_coll,
    xml_file_xstr            TYPE xstring.

  add_info_mesg 'Start to collect DPT Usage data'.

  "Check buffer
  lcl_data_collection_helper=>is_data_in_buffer_old(
     EXPORTING
       iv_trigid             = c_data_key-data_trigid
       iv_trigoffset         = c_data_key-data_trigoffset
       iv_subid              = subid_dpt
       iv_client_dependent   = abap_true
     IMPORTING
       ev_data_in_buffer_is_old = lv_data_in_buffer_is_old
       ev_buffered_data         = xml_file_xstr
   ).

  IF lv_data_in_buffer_is_old <> abap_true.

  ELSE.
    CREATE OBJECT lo_dpt_api TYPE lcl_dpt_data_coll.

    CLEAR xml_file_xstr.
    xml_file_xstr = lo_dpt_api->collect_data( ).

    IF xml_file_xstr IS NOT INITIAL.
      lcl_data_collection_helper=>add_data_2_buffer(
        EXPORTING
          iv_trigid           = c_data_key-data_trigid
          iv_trigoffset       = c_data_key-data_trigoffset
          iv_subid            = subid_dpt
          iv_data_2_buffer    = xml_file_xstr
          iv_client_dependent = abap_true
      ).
    ENDIF.
  ENDIF.

  IF xml_file_xstr IS INITIAL.
    " string is empty because no / not enough data are available
     add_info_mesg 'DPT Usage data not available'.
  ELSE.
    "Add xml data to the ZIP file
    cl_tmw_rc_collect_bpa=>so_data_file->add(
      name    = 'usage_data.xml'
      content = xml_file_xstr ).

    add_info_mesg 'DPT Usage data is collected'.
  ENDIF.
ENDFORM.                    " COLLECT_USAGE_DATA
