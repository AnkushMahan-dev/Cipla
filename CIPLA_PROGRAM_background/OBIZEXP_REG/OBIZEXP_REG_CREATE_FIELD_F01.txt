FORM events_build USING lt_events TYPE slis_t_event.

  DATA: ls_event TYPE slis_alv_event.

*---------------- RETURNS TABLE OF POSSIBLE EVENTS FOR A LIST TYPE----*
  CLEAR   lt_events.
  REFRESH lt_events.

  CALL FUNCTION 'REUSE_ALV_EVENTS_GET'
    EXPORTING
      i_list_type     = 0
    IMPORTING
      et_events       = lt_events
    EXCEPTIONS
      list_type_wrong = 1
      OTHERS          = 2.

  IF sy-subrc NE 0.
    CLEAR   lt_events.
    REFRESH lt_events.
  ENDIF.

*--------------------TOP OF PAGE EVENT--------------------------------*
  READ TABLE lt_events INTO ls_event WITH KEY name = 'TOP_OF_PAGE'.
  MOVE g_form_top_of_page TO ls_event-form.
  MODIFY lt_events FROM ls_event INDEX sy-tabix.

ENDFORM.                    " events_build

*-----------------TOP OF THE PAGE -->DYNAMIC CALL--------------------*
FORM top_of_page.

  DATA: it_top TYPE slis_t_listheader,
        wa_top TYPE slis_listheader.

  DATA: wrk_dtlow  TYPE string,
        wrk_dthigh TYPE string.

  CLEAR wa_top.
  wa_top-typ = 'H'.
  wa_top-info = 'Export Register'.
  APPEND wa_top TO it_top.

  CLEAR wa_top.
  wa_top-typ = 'S'.

  CONCATENATE s_date-low+6(2) '.'
              s_date-low+4(2) '.'
              s_date-low(4) INTO wrk_dtlow.

  CONCATENATE s_date-high+6(2) '.'
              s_date-high+4(2) '.'
              s_date-high(4) INTO wrk_dthigh.

  CONCATENATE 'From' wrk_dtlow 'To' wrk_dthigh
              INTO wa_top-info SEPARATED BY space.

  APPEND wa_top TO it_top.

  CALL FUNCTION 'REUSE_ALV_COMMENTARY_WRITE'
    EXPORTING
      it_list_commentary = it_top.

ENDFORM.                    "TOP_OF_PAGE
