        FROM mast CLIENT SPECIFIED

          wa_download-ncost = 'X'.
        ENDIF.
*        wa_download-ncost = wa_marc_dwn-ncost.
      ENDIF.

      CLEAR wa_mast_dwn.
      READ TABLE ta_mast_dwn INTO wa_mast_dwn WITH KEY matnr = wa_keko_dwn-matnr werks = wa_keko_dwn-werks.
      IF sy-subrc = 0.
        CLEAR wa_stpo_dwn.
        READ TABLE ta_stpo_dwn INTO wa_stpo_dwn WITH KEY stlnr = wa_mast_dwn-stlnr.

      ENDIF.

    IF sy-subrc = 0.
      wa_download-kltext = wa_csku-ltext.
