    ENDIF.
*   determine work process

  ENDIF.

* choose relative Max. No. Processes ?
  CHECK NOT pa_rel IS INITIAL.

  IF pa_rel_v < 1 OR pa_rel_v > 100.
    SET CURSOR FIELD 'PA_REL_V'.
    MESSAGE e008.
  ENDIF.
  IF sy-ucomm IS INITIAL.
*   press return-button -> only check if Server Group is available
    IF NOT pa_srgrp IS INITIAL.
      SELECT * FROM rzllitab UP TO 1 ROWS INTO TABLE lt_server_groups
           WHERE classname =  pa_srgrp
           AND   grouptype <> space.
      IF sy-dbcnt = 0.
        SET CURSOR FIELD 'PA_SRGRP'.
        MESSAGE e831(bt) WITH pa_srgrp.
      ENDIF.   "sy-dbcnt = 0.
    ENDIF.   "NOT pa_srgrp IS INITIAL.
  ELSE.  "sy-ucomm IS INITIAL.
*   check Server Group and determine work process of Server Group
    ls_parpar-serv_group = pa_srgrp.
*   In case of resource failure try again but only for
*   one hour to avoid that system is blocked in total
    CLEAR lv_tries.
    DO.
    lv_tries = lv_tries + 1.
    CALL FUNCTION 'SPBT_INITIALIZE'
      EXPORTING
        group_name                     = ls_parpar-serv_group
      IMPORTING
        max_pbt_wps                    = lv_max_pbt_wps
      EXCEPTIONS
        invalid_group_name             = 1
        internal_error                 = 2
        pbt_env_already_initialized    = 3
        currently_no_resources_avail   = 4
        no_pbt_resources_found         = 5
        cant_init_different_pbt_groups = 6
        OTHERS                         = 7.
    CASE sy-subrc.
      WHEN 0.
        EXIT.
*       Everything is alright!
      WHEN 3.
*       Server group already initialized. Get the current res. info...
        CALL FUNCTION 'SPBT_GET_CURR_RESOURCE_INFO'
          IMPORTING
            max_pbt_wps                 = lv_max_pbt_wps
          EXCEPTIONS
            internal_error              = 1
            pbt_env_not_initialized_yet = 2
            OTHERS                      = 3.
        IF sy-subrc <> 0.
*         Error
          SET CURSOR FIELD 'PA_SRGRP'.
          MESSAGE ID sy-msgid
            TYPE 'E' NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
        ELSE.
          EXIT.
        ENDIF.
      WHEN 4.
*       Stop processing when since one hour no resource
*       is available
        IF lv_tries > 350.
          SET CURSOR FIELD 'PA_SRGRP'.
          MESSAGE ID sy-msgid
            TYPE 'E' NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
        ELSE.
          WAIT UP TO 10 SECONDS.
        ENDIF.
      WHEN OTHERS.
*       Error
        SET CURSOR FIELD 'PA_SRGRP'.
        MESSAGE ID sy-msgid
          TYPE 'E' NUMBER sy-msgno
          WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
    ENDCASE.
    ENDDO.
*   Determine work process
    ls_parpar-max_workp = ( lv_max_pbt_wps * pa_rel_v ) / 100.
    IF ls_parpar-max_workp < 1.
      ls_parpar-max_workp = 1.
    ENDIF.    "ls_parpar-max_workp < 1.
  ENDIF.   "sy-ucomm IS INITIAL.

                AND status = gc_cif_imstat_n_con    " Keine Deltamodelle
             ORDER BY modelname logsys apoapp
                      crdate DESCENDING crtime DESCENDING.
            LOOP AT lt_cif_imod_deact.
              PERFORM model_output
              USING lt_cif_imod_deact
              icon_cancel.
            ENDLOOP.
          ELSE.
*       sowieso aktive nur ausgeben.
            PERFORM model_output
                USING
                  ls_cif_imod
                  icon_okay.
          ENDIF.
        ELSE.
*    es soll deaktiviert werden, also die aktiven auf inaktiv
*    setzen und ausgeben
          IF ls_cif_imod-active <> gc_active-inactive.
            ls_cif_imod-active = gc_active-inactive.
            MODIFY lt_cif_imod INDEX sy-tabix
                 FROM ls_cif_imod TRANSPORTING active.
*         als zu deaktivieren ausgeben
            PERFORM model_output
                USING
                  ls_cif_imod
                  icon_cancel.
          ELSE.
*         als unverändert ausgeben
            PERFORM model_output
                USING
                  ls_cif_imod
                  space.
          ENDIF.
        ENDIF.
      ENDAT.

      IF NOT l_newest = g_true_con.
        IF ls_cif_imod-active <> gc_active-inactive.
          ls_cif_imod-active = gc_active-inactive.
          MODIFY lt_cif_imod INDEX sy-tabix
               FROM ls_cif_imod TRANSPORTING active.
          PERFORM model_output
              USING
                ls_cif_imod
                icon_cancel.
        ELSE.
          PERFORM model_output
              USING
                ls_cif_imod
                space.
        ENDIF.
      ENDIF.

      index_relation-table_index = l_table_index.
      index_relation-line_index  = sy-linno.
      APPEND index_relation.

    ENDLOOP.

    ULINE.


***********************************************************************
***********************************************************************
* Alle Modell abgehen und die Filterobjekte abhandeln
***********************************************************************
***********************************************************************

    LOOP AT lt_cif_imod.

* Konsistenz überprüfen
      PERFORM imodel_check
         USING lt_cif_imod.

    ENDLOOP.

    SET PF-STATUS 'S101'.

    IF NOT sy-batch IS INITIAL OR NOT dark IS INITIAL AND lv_workflow IS INITIAL.
      PERFORM models_activate.
    ENDIF.
  ENDIF.
***********************************************************************

AT USER-COMMAND.

  CASE sy-ucomm.

    WHEN 'ACTV'.
      PERFORM models_activate.
*      message id g_msgid_cif_con type g_msgtyp_i_con number 030.
      PERFORM message_or_log
        USING
          dark
          g_msgid_cif_con
          g_msgtyp_i_con
          030
          space space space space.

  ENDCASE.

FORM models_activate.

*----------------------------------------------------------------------
* Aktivierung bzw. Deaktivierung der gewählten Modelle
  DATA: lt_logsys LIKE cif_dest OCCURS 0 WITH HEADER LINE,
        lt_change_models LIKE cif_imod OCCURS 0 WITH HEADER LINE,
        cv_incerr TYPE xfeld.

  LOOP AT lt_cif_imod.
    lt_logsys-logsys = lt_cif_imod-logsys.
    APPEND lt_logsys.
  ENDLOOP.

* logische Systeme zusammenfassen
  SORT lt_logsys BY logsys.
  DELETE ADJACENT DUPLICATES FROM lt_logsys.

  LOOP AT lt_logsys.

    CLEAR lt_change_models.
    REFRESH lt_change_models.

    IF  i_active = g_true_con.
* Es sollen Modelle aktiviert werden
      LOOP AT lt_cif_imod WHERE logsys = lt_logsys AND active <> space.
* Alle Typ/Segmentkombinationen des zu aktivierenden Modells selektieren
        SELECT * FROM cif_imod APPENDING TABLE lt_change_models
               WHERE modelname = lt_cif_imod-modelname
                 AND logsys    = lt_cif_imod-logsys
                 AND apoapp    = lt_cif_imod-apoapp
                 AND crdate    = lt_cif_imod-crdate
                 AND crtime    = lt_cif_imod-crtime
               ORDER BY imtyp segment.
* Das neue Aktiv-Flag setzen
        lt_change_models-active = gc_active-incr.
        MODIFY lt_change_models TRANSPORTING active
                WHERE modelname = lt_cif_imod-modelname
                  AND logsys    = lt_cif_imod-logsys
                  AND apoapp    = lt_cif_imod-apoapp
                  AND crdate    = lt_cif_imod-crdate
                  AND crtime    = lt_cif_imod-crtime.
*     Bei den zu aktivierenden eventuell andere aktive 'Versionen'
*     deaktivieren
        SELECT * FROM cif_imod APPENDING TABLE lt_change_models
               WHERE modelname = lt_cif_imod-modelname
                 AND logsys    = lt_cif_imod-logsys
                 AND apoapp    = lt_cif_imod-apoapp
                 AND  (   crdate    <> lt_cif_imod-crdate
                       OR crtime    <> lt_cif_imod-crtime )
                 AND active   <> space
               ORDER BY imtyp segment.
        IF sy-dbcnt > 0.
          lt_change_models-active = gc_active-inactive.
          MODIFY lt_change_models TRANSPORTING active
                  WHERE modelname = lt_cif_imod-modelname
                    AND logsys    = lt_cif_imod-logsys
                    AND apoapp    = lt_cif_imod-apoapp
                   AND  (   crdate    <> lt_cif_imod-crdate
                         OR crtime    <> lt_cif_imod-crtime ).
        ENDIF.
      ENDLOOP.
    ELSE.
* es sollen alle gefundenen aktiven deaktiviert werden
      LOOP AT lt_cif_imod WHERE logsys = lt_logsys.
*       Alle aktiven Typ/Segmentkombinationen des eventuell zu
*       deaktivierenden Modells selektieren
        SELECT * FROM cif_imod APPENDING TABLE lt_change_models
               WHERE modelname = lt_cif_imod-modelname
                 AND logsys    = lt_cif_imod-logsys
                 AND apoapp    = lt_cif_imod-apoapp
                 AND crdate    = lt_cif_imod-crdate
                 AND crtime    = lt_cif_imod-crtime
                 AND active    <> space
               ORDER BY imtyp segment.
        IF sy-dbcnt > 0.
          lt_change_models-active = gc_active-inactive.
          MODIFY lt_change_models TRANSPORTING active
                  WHERE modelname = lt_cif_imod-modelname
                    AND logsys    = lt_cif_imod-logsys
                    AND apoapp    = lt_cif_imod-apoapp
                    AND crdate    = lt_cif_imod-crdate
                    AND crtime    = lt_cif_imod-crtime.
        ENDIF.
      ENDLOOP.
    ENDIF.

    IF NOT lt_change_models[] IS INITIAL.

* check settings for class system
      CALL FUNCTION 'CIF_CHECK_CFGSYS'
        EXPORTING
          iv_logsys          = lt_logsys-logsys
          iv_cfgsys          = lv_cfgsys
          iv_conflict_action = p_cnflct
        IMPORTING
          ev_cfgsys_force    = lv_cfgsys_force
        TABLES
          it_change_models   = lt_change_models
        EXCEPTIONS
          abort_initialload  = 1
          OTHERS             = 2.
      IF NOT sy-subrc IS INITIAL.
        CONTINUE.
      ENDIF.

      CALL FUNCTION 'CIF_ADC_TRANSMIT_ALL'
        EXPORTING
          iv_modus           = 'I'
          iv_ignore_fqentry  = ignfqety
          iv_deact_proto     = p_deactp
          iv_act_proto       = p_actp
          is_parpar          = ls_parpar
          iv_snp_plaf        = snp_plaf
          iv_cfgsys          = lv_cfgsys
          iv_cfgsys_force    = lv_cfgsys_force
          iv_locbp           = p_locbp
          iv_no_queuewarning = nopcwarn
          iv_incerr          = p_incerr
        TABLES
          change_models      = lt_change_models
        CHANGING
          cv_incerr          = cv_incerr
        EXCEPTIONS
          error_message      = 98
          different_logsys   = 1
          internal_error     = 3
          OTHERS             = 4.

      CASE sy-subrc.
        WHEN 0.

        WHEN OTHERS.
*          message id sy-msgid type sy-msgty number sy-msgno
*                  with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
          PERFORM write_result.
          PERFORM message_or_log
             USING
               dark
               sy-msgid sy-msgty sy-msgno
               sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
      ENDCASE.
      IF cv_incerr IS NOT INITIAL.
        MESSAGE ID 'XC' TYPE 'E' NUMBER '000'.
      ENDIF.
    ENDIF.
  ENDLOOP.

ENDFORM.                    "MODELS_ACTIVATE

FORM imodel_check USING p_cif_imod LIKE cif_imod.

* check for active/inactive mix
  DATA: lt_active LIKE cif_imod OCCURS 0 WITH HEADER LINE,
        l_inactive_lines TYPE i,
        l_active_lines TYPE i.

  SELECT DISTINCT active FROM cif_imod
             INTO CORRESPONDING FIELDS OF TABLE lt_active
             WHERE modelname = p_cif_imod-modelname
               AND logsys = p_cif_imod-logsys
               AND apoapp = p_cif_imod-apoapp
               AND cruser = p_cif_imod-cruser
               AND crdate = p_cif_imod-crdate
               AND crtime = p_cif_imod-crtime.
  LOOP AT lt_active.
    IF lt_active-active = gc_active-inactive.
      ADD 1 TO l_inactive_lines.
    ELSE.
      ADD 1 TO l_active_lines.
    ENDIF.
  ENDLOOP.
  IF l_active_lines > 0 AND l_inactive_lines > 0.
*    message i036
*        with p_cif_imod-modelname p_cif_imod-logsys p_cif_imod-apoapp.
    PERFORM message_or_log
       USING
         dark
         g_msgid_cif_con g_msgtyp_i_con 036
         p_cif_imod-modelname p_cif_imod-logsys p_cif_imod-apoapp space.
  ENDIF.
ENDFORM.                               " IMODEL_CHECK

*---------------------------------------------------------------------*

FORM model_output
     USING
       ps_cif_imod LIKE cif_imod
       p_icon      LIKE icon_okay.

  IF NOT p_icon IS INITIAL.
    l_icon = p_icon.
    WRITE: /  l_icon AS ICON.
  ELSE.
    CLEAR l_icon.
    WRITE: / l_icon.
  ENDIF.

  WRITE:      ps_cif_imod-modelname UNDER text-001,
              ps_cif_imod-logsys UNDER text-002,
              ps_cif_imod-apoapp UNDER text-003,
              ps_cif_imod-crdate UNDER text-007,
              ps_cif_imod-crtime,
              ps_cif_imod-cruser UNDER text-004.

ENDFORM.                    "MODEL_OUTPUT

*       FORM WRITE_RESULT                                             *
*---------------------------------------------------------------------*

FORM write_result.

  DATA: lv_really_active LIKE cif_imod-active.
  DATA: l_temp_icon LIKE l_icon.       " Jaja, ABAP...

  LOOP AT index_relation.

*   go get integration model details
    READ TABLE lt_cif_imod INDEX index_relation-table_index
          INTO ls_cif_imod.

*   go get display details
    READ LINE index_relation-line_index.

*   make sure we know the 'real' current activation status since
*   there was a loop over the logical systems and some systems
*   might have been processed without errors
    SELECT SINGLE active INTO lv_really_active FROM cif_imod
      WHERE modelname = ls_cif_imod-modelname
        AND logsys    = ls_cif_imod-logsys
        AND apoapp    = ls_cif_imod-apoapp
        AND crdate    = ls_cif_imod-crdate
        AND crtime    = ls_cif_imod-crtime.                 "#EC *

*   if the activation status does not match what we were trying to do
    IF ( NOT ls_cif_imod-active IS INITIAL
         AND lv_really_active IS INITIAL )
    OR ( ls_cif_imod-active IS INITIAL
         AND NOT lv_really_active IS INITIAL ).

*     display as failure
      l_temp_icon = icon_failure.
      l_icon = l_temp_icon.
      icon_prepare_for_modify l_temp_icon.
      MODIFY LINE index_relation-line_index
            FIELD VALUE l_icon FROM l_temp_icon.

    ENDIF.

  ENDLOOP.

ENDFORM.                    "WRITE_RESULT
