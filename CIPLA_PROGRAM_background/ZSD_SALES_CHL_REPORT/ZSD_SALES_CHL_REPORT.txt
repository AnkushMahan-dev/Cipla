FORM form_break_qty_into_cases_unit .

  LOOP AT it_alv3.

    DATA: l_meins TYPE meins.
    CLEAR l_meins.

    SELECT SINGLE meins FROM mara INTO l_meins WHERE matnr = it_alv3-matnr.

    IF l_meins NE 'CS'.
      SELECT SINGLE *
        FROM marm
      WHERE meinh EQ it_alv3-vrkme      "'CS'
      AND matnr EQ it_alv3-matnr.
    ELSE.
      SELECT SINGLE *
      FROM marm
      WHERE meinh NE 'CS'
      AND matnr EQ it_alv3-matnr.

    ENDIF.

    IF sy-subrc EQ 0.
      IF l_meins NE 'CS'.
        it_alv3-fkimg1 = marm-umrez * it_alv3-fkimg.
      ELSE.
        it_alv3-fkimg1 = marm-umren * it_alv3-fkimg.
      ENDIF.
      DATA str0    TYPE string.
      DATA str1    TYPE string.
      DATA: val     TYPE c .
      DATA str2(3) TYPE n.

      MOVE it_alv3-fkimg1 TO str0.
      SPLIT str0 AT '.' INTO str1 str2.
      val = str2+0(1) .
      IF val GE 5 .
        COMPUTE it_alv3-fkimg1 = ceil( it_alv3-fkimg1 ).
      ELSE.
        COMPUTE it_alv3-fkimg1 = floor( it_alv3-fkimg1 ).
      ENDIF.
*      IT_ALV3-VRKME  = MARM-MEINH.

      IF it_alv3-vrkme = 'CS'.
        it_alv3-vrkme1 = 'EA'.
      ELSE.
*        IT_ALV3-VRKME1 = IT_ALV3-VRKME.
        it_alv3-vrkme1 = marm-meinh.
      ENDIF.

      MODIFY it_alv3.
    ELSE.
      it_alv3-fkimg1 = it_alv3-fkimg.
      it_alv3-fkimg  = ''.
      it_alv3-vrkme1 = it_alv3-vrkme.
      it_alv3-vrkme  = ''.
      MODIFY it_alv3.
    ENDIF.

***    IF  IT_VBRP-FKART EQ 'RE' OR IT_VBRP-FKART = 'ZRE'.
***      IT_ALV3-FKIMG   = IT_ALV3-FKIMG   *  '-1'.
***      IT_ALV3-FKIMG1  = IT_ALV3-FKIMG1  *  '-1'.
***    ENDIF.
    MODIFY it_alv3.
  ENDLOOP.
