FORM get_ekbe .

  IF ta_ekko_ekpo_eket[] IS NOT INITIAL.

    SORT ta_ekko_ekpo_eket BY ebeln ebelp.

    SELECT ebeln
           ebelp
           gjahr
           bewtp
           belnr
           buzei
           bwart
           budat
           menge
           shkzg
           charg
           FROM ekbe
           INTO  CORRESPONDING FIELDS OF TABLE ta_ekbe
           FOR ALL ENTRIES IN ta_ekko_ekpo_eket
           WHERE ebeln EQ ta_ekko_ekpo_eket-ebeln
    AND ebelp EQ ta_ekko_ekpo_eket-ebelp.
  ENDIF.
  IF ta_ekbe[] IS NOT INITIAL.

    SORT ta_ekbe BY bewtp.

    DELETE ta_ekbe WHERE bewtp NE 'E'.


    SORT ta_ekbe BY ebeln ebelp.

    LOOP AT ta_ekko_ekpo_eket INTO wa_ekko_ekpo_eket.

      wa_objectid-sign = 'I'.
      wa_objectid-option ='EQ'.
      MOVE wa_ekko_ekpo_eket-banfn TO wa_objectid-low.
      APPEND wa_objectid TO ta_objectid.
      CLEAR: wa_objectid.

      wa_objectid-sign = 'I'.
      wa_objectid-option ='EQ'.
      MOVE wa_ekko_ekpo_eket-ebeln TO wa_objectid-low.
      APPEND wa_objectid TO ta_objectid.
      CLEAR: wa_objectid.

      CLEAR : wa_ekbe.
      READ TABLE ta_ekbe  INTO wa_ekbe  WITH KEY ebeln = wa_ekko_ekpo_eket-ebeln
                                                 ebelp = wa_ekko_ekpo_eket-ebelp
                                                 BINARY SEARCH.
      IF sy-subrc EQ 0.

      ELSE.
        wa_ekbe-ebeln = wa_ekko_ekpo_eket-ebeln.
        wa_ekbe-ebelp = wa_ekko_ekpo_eket-ebelp.

        APPEND wa_ekbe TO ta_ekbe1.

      ENDIF.

      CLEAR :wa_ekko_ekpo_eket,wa_ekbe.
    ENDLOOP.

    SORT ta_objectid BY low.
    DELETE ADJACENT DUPLICATES FROM ta_objectid COMPARING low.
  ELSE.

    LOOP AT ta_ekko_ekpo_eket INTO wa_ekko_ekpo_eket.
      wa_objectid-sign = 'I'.
      wa_objectid-option ='EQ'.
      MOVE wa_ekko_ekpo_eket-banfn TO wa_objectid-low.
      APPEND wa_objectid TO ta_objectid.
      CLEAR: wa_objectid.

      wa_objectid-sign = 'I'.
      wa_objectid-option ='EQ'.
      MOVE wa_ekko_ekpo_eket-ebeln TO wa_objectid-low.
      APPEND wa_objectid TO ta_objectid.
      CLEAR: wa_objectid.

      wa_ekbe-ebeln = wa_ekko_ekpo_eket-ebeln.
      wa_ekbe-ebelp = wa_ekko_ekpo_eket-ebelp.

      APPEND wa_ekbe TO ta_ekbe1.
      CLEAR :wa_ekko_ekpo_eket,wa_ekbe.
    ENDLOOP.

    SORT ta_objectid BY low.
    DELETE ADJACENT DUPLICATES FROM ta_objectid COMPARING low.

  ENDIF.

  IF ta_ekbe1[] IS NOT INITIAL.

    LOOP AT ta_ekbe1 INTO wa_ekbe1.
      APPEND wa_ekbe1 TO ta_ekbe.
      CLEAR: wa_ekbe1.
    ENDLOOP.
  ENDIF.
ENDFORM.                    " GET_EKBE

FORM qc_per .

         l_tot_qc  TYPE vtbbewe-atage.

  PERFORM qc_leadtime.

  ta_qals_t[] = ta_qals[].

  DESCRIBE TABLE ta_qals_a LINES l_acc.
  DESCRIBE TABLE ta_qals_r LINES l_rej.

  IF l_total IS NOT INITIAL.
    wa_final-avg_qcper = ( l_rej / l_total ) * 100.
    wa_final-avg_qc1 = wa_final-avg_qc / l_total.

  ENDIF.

FORM qc_leadtime .

  IF wa_ekko_ekpo_eket-ebeln IS NOT INITIAL.



    ta_mseg_t[] = ta_mseg[].
*  ta_mseg1_t[] = ta_mseg[].
*  ta_mseg2_t[] = ta_mseg[].
*  ta_mseg3_t[] = ta_mseg[].

    SORT ta_mseg_t BY ebeln ebelp.
    DELETE ta_mseg_t WHERE ebeln NE wa_ekko_ekpo_eket-ebeln
                     AND    ebelp NE wa_ekko_ekpo_eket-ebelp.




    SORT ta_mseg_t BY matnr werks charg ebeln ebelp.
    DELETE ADJACENT DUPLICATES FROM ta_mseg_t COMPARING matnr werks charg ebeln ebelp.


    LOOP AT ta_mseg1 INTO wa_mseg1 WHERE ebeln = wa_ekko_ekpo_eket-ebeln
                                   AND   ebelp = wa_ekko_ekpo_eket-ebelp.
*    CLEAR :wa_mseg1.
      READ TABLE ta_mseg1 INTO wa_mseg1 WITH KEY ebeln = wa_ekko_ekpo_eket-ebeln
                                                 ebelp = wa_ekko_ekpo_eket-ebelp
                                                 charg = wa_mseg1-charg.

      IF sy-subrc EQ 0.
        CLEAR :wa_mseg2.
        READ TABLE ta_mseg2 INTO wa_mseg2 WITH KEY ebeln = wa_ekko_ekpo_eket-ebeln
                                                   ebelp = wa_ekko_ekpo_eket-ebelp
                                                   charg = wa_mseg1-charg.

        IF sy-subrc EQ 0.

        ELSE.
          CLEAR :wa_mseg3.
          READ TABLE ta_mseg3 INTO wa_mseg3 WITH KEY ebeln = wa_ekko_ekpo_eket-ebeln
                                                     ebelp = wa_ekko_ekpo_eket-ebelp
                                                     charg = wa_mseg1-charg.
          IF sy-subrc EQ 0.

            wa_mseg2-budat_mkpf = wa_mseg3-budat_mkpf.
          ENDIF.
          CLEAR wa_mseg4.
          READ TABLE ta_mseg4 INTO wa_mseg4 WITH KEY ebeln = wa_ekko_ekpo_eket-ebeln
                                                     ebelp = wa_ekko_ekpo_eket-ebelp
                                                     charg = wa_mseg1-charg.
          IF sy-subrc EQ 0.
            wa_mseg2-budat_mkpf = wa_mseg-budat_mkpf.
          ENDIF.
        ENDIF.


        DATA : l_avg_qc1 TYPE vtbbewe-atage,
               l_tot_qc  TYPE vtbbewe-atage.

        PERFORM cal_time USING wa_mseg1-budat_mkpf
                               wa_mseg2-budat_mkpf
                               CHANGING l_avg_qc1.

        IF wa_mseg2-budat_mkpf LT wa_mseg1-budat_mkpf.
          l_avg_qc1 = l_avg_qc1 * -1.
        ENDIF.
        wa_final-avg_qc = l_avg_qc1 + wa_final-avg_qc.

      ENDIF.

      CLEAR :wa_mseg_t,l_avg_qc1.
    ENDLOOP.




  ENDIF.

FORM qty_cal .


  DATA : l_ekbeqty  TYPE eket-menge,
         l_eketqty  TYPE eket-menge,
         l_pekbeqty TYPE eket-menge,
         l_peketqty TYPE eket-menge.

  LOOP AT ta_ekko_ekpo_eket INTO wa_ekko_ekpo_eket.

    CONCATENATE  wa_ekko_ekpo_eket-ebeln
                 wa_ekko_ekpo_eket-ebelp
                 INTO wa_ekko_ekpo_eket-eket_key.



    MODIFY ta_ekko_ekpo_eket FROM wa_ekko_ekpo_eket TRANSPORTING eket_key .



    CLEAR :wa_ekko_ekpo_eket.

  ENDLOOP.

  SORT ta_ekko_ekpo_eket BY ebeln ebelp eindt ASCENDING.

  LOOP AT ta_ekko_ekpo_eket INTO wa_ekko_ekpo_eket.


    ON CHANGE OF  wa_ekko_ekpo_eket-eket_key.
      CLEAR : l_eketqty.
    ENDON.

    l_eketqty = l_eketqty + wa_ekko_ekpo_eket-menge.

    wa_ekko_ekpo_eket-cum_eket_qty =  l_eketqty.


    MODIFY ta_ekko_ekpo_eket FROM wa_ekko_ekpo_eket TRANSPORTING  cum_eket_qty.

    CLEAR :wa_ekko_ekpo_eket.

  ENDLOOP.




  LOOP AT ta_ekbe_temp INTO wa_ekbe_temp.

    CONCATENATE  wa_ekbe_temp-ebeln
                 wa_ekbe_temp-ebelp
                   INTO wa_ekbe_temp-ekbe_key.


    MODIFY ta_ekbe_temp FROM wa_ekbe_temp TRANSPORTING ekbe_key.

    CLEAR :wa_ekbe_temp.


  ENDLOOP.



  SORT ta_ekbe_temp BY ebeln ebelp budat  ASCENDING.

  LOOP AT ta_ekbe_temp INTO wa_ekbe_temp.


    ON CHANGE OF  wa_ekbe_temp-ekbe_key.
      CLEAR : l_ekbeqty.
    ENDON.

    l_ekbeqty = l_ekbeqty + wa_ekbe_temp-menge.

    wa_ekbe_temp-cum_ekbe_qty =  l_ekbeqty.


    MODIFY ta_ekbe_temp FROM wa_ekbe_temp TRANSPORTING  cum_ekbe_qty.

    CLEAR :wa_ekbe_temp.

  ENDLOOP.
