****
****    IF  FINAL-LOW = 'X'.
****      CONCATENATE LV_WHERE_CLAUSE  ' PACKNO IN ( select HPACKNO from ESKL where FINAL = ''X'' ) ' INTO LV_WHERE_CLAUSE.
****    ENDIF.

  IF P_BANF IS NOT INITIAL.
    CONCATENATE LV_WHERE_CLAUSE '''' 'B' '''' INTO LV_WHERE_CLAUSE.
    LV_FLAG = 'X'.
  ENDIF.

  IF P_ANFR IS NOT INITIAL.
    IF LV_FLAG EQ 'X'.
      CONCATENATE LV_WHERE_CLAUSE ',' INTO LV_WHERE_CLAUSE.
    ENDIF.
    CONCATENATE LV_WHERE_CLAUSE '''' 'A' '''' INTO LV_WHERE_CLAUSE.
    LV_FLAG = 'X'.
  ENDIF.

  IF P_BEST IS NOT INITIAL.
    IF LV_FLAG EQ 'X'.
      CONCATENATE LV_WHERE_CLAUSE ',' INTO LV_WHERE_CLAUSE.
    ENDIF.
    CONCATENATE LV_WHERE_CLAUSE '''' 'F' '''' INTO LV_WHERE_CLAUSE.
    LV_FLAG = 'X'.
  ENDIF.

  IF P_KONTR IS NOT INITIAL.
    IF LV_FLAG EQ 'X'.
      CONCATENATE LV_WHERE_CLAUSE ',' INTO LV_WHERE_CLAUSE.
    ENDIF.
    CONCATENATE LV_WHERE_CLAUSE '''' 'K' '''' INTO LV_WHERE_CLAUSE.
    LV_FLAG = 'X'.
  ENDIF.

  IF P_ERFBL IS NOT INITIAL.
    IF LV_FLAG EQ 'X'.
      CONCATENATE LV_WHERE_CLAUSE ',' INTO LV_WHERE_CLAUSE.
    ENDIF.
    CONCATENATE LV_WHERE_CLAUSE  '''' 'Q' '''' INTO LV_WHERE_CLAUSE.
  ENDIF.

  CONCATENATE LV_WHERE_CLAUSE ')' INTO LV_WHERE_CLAUSE.
*********************************************************************************
DATA: lv_string TYPE string.
IF lv_where_clause eq 'srvpos IN srvpos AND matkl IN srvmatkl AND ebeln IN ebeln AND bstyp IN ()'.
  lv_string = substring_before( val = lv_where_clause sub = 'AND bstyp').
  lv_where_clause = lv_string.
ENDIF.

  REFRESH XDATA.
  IF KTEXT1 IS INITIAL.

    SELECT * FROM ML_ESLL INTO TABLE XML_ESLL BYPASSING BUFFER

                                 AND PACKAGE EQ SPACE.

  ELSE.
    REFRESH KTEXT1_RANGE.
    CLEAR KTEXT1_RANGE.

    MOVE 'CP' TO KTEXT1_RANGE-OPTION.
    MOVE 'I'  TO KTEXT1_RANGE-SIGN.

****    IF  FINAL-LOW = 'X'.
****      CONCATENATE LV_WHERE_CLAUSE  ' AND PACKNO IN ( select HPACKNO from ESKL where FINAL = ''X'' ) ' INTO LV_WHERE_CLAUSE.
****    ENDIF.

FORM CHANGE_LAYOUT .
  LOOP AT SCREEN.

    CASE SCREEN-GROUP1.
      WHEN 'GR1'.
        IF RF_SWITCH EQ ABAP_TRUE.
          SCREEN-ACTIVE = 0.
          MODIFY SCREEN.
        ENDIF.
      WHEN 'GR2'.
        IF RF_SWITCH EQ ABAP_FALSE.
          SCREEN-ACTIVE = 0.
          MODIFY SCREEN.
        ENDIF.
    ENDCASE.

  ENDLOOP.


ENDFORM.                    " CHANGE_LAYOUT
