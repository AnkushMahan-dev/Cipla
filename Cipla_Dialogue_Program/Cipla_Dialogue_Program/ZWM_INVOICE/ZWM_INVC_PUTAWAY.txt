REPORT zwm_invc_putaway.
TABLES : vbrk.

TYPES: BEGIN OF ty_final,
         chk_box    TYPE char1,
         lgnum      TYPE lagp-lgnum,
         likp_vbeln TYPE likp-vbeln,
         vbeln      TYPE vbrk-vbeln,
         erdat      TYPE vbrk-erdat,
         kunrg      TYPE vbrk-kunrg,
         name1      TYPE kna1-name1,

       ls_final TYPE ty_final.

DATA : gt_zwm_invc_put TYPE TABLE OF zwm_invc_put,
       gt_fieldcat     TYPE slis_t_fieldcat_alv.

PARAMETERS : p_lgnum TYPE lgnum OBLIGATORY.
SELECT-OPTIONS : s_fkart FOR vbrk-fkart OBLIGATORY,
                 s_erdat FOR vbrk-erdat OBLIGATORY,
                 s_vbeln FOR vbrk-vbeln.
PARAMETERS : p_werks TYPE werks_d OBLIGATORY.
SELECTION-SCREEN : END OF BLOCK b1.


AT SELECTION-SCREEN.
  "Plant Authority check
  AUTHORITY-CHECK OBJECT 'ZSD_WERKS'
         ID 'WERKS' FIELD  p_werks.
  IF sy-subrc <> 0.
    MESSAGE e084(zsd) WITH  p_werks.
  ENDIF.

  SELECT SINGLE * FROM t320
  INTO @DATA(ls_t320)
  WHERE werks = @p_werks.

  IF p_lgnum IS NOT INITIAL.
    IF sy-subrc = 0.
      IF ls_t320-lgnum NE p_lgnum.
        MESSAGE 'Invalid Warehouse' TYPE 'E'.
      ENDIF.
    ENDIF.
  ENDIF.

START-OF-SELECTION.

  SELECT *
  FROM vbrk
  INTO TABLE @DATA(lt_vbrk)
  WHERE erdat IN @s_erdat
  AND vbeln IN @s_vbeln
  AND fkart IN @s_fkart "NE 'S1'
  AND fksto NE 'X'.
*
*  DATA(lt_cancel) = lt_vbrk.
*  DELETE lt_cancel WHERE fkart NE 'S1'.

  IF lt_vbrk IS NOT INITIAL.

    SELECT *
    FROM vbrp
    INTO TABLE @DATA(lt_vbrp)

    WHERE vbeln EQ @lt_vbrk-vbeln.

    IF lt_vbrp IS NOT INITIAL.

      SELECT *
      FROM kna1
      INTO TABLE @DATA(lt_kna1)
      FOR ALL ENTRIES IN @lt_vbrk
      WHERE kunnr = @lt_vbrk-kunrg.

      SELECT *
      FROM zwm_invc_put
      INTO TABLE gt_zwm_invc_put
      FOR ALL ENTRIES IN lt_vbrp
      WHERE lgnum EQ p_lgnum
      AND delivery_no EQ lt_vbrp-vgbel
      AND chk_box NE 'X'.

      SELECT * FROM vbuk
      INTO TABLE @DATA(lt_vbuk)
      FOR ALL ENTRIES IN @lt_vbrp
      WHERE vbeln = @lt_vbrp-vgbel
      AND fkstk = 'C'.

    ENDIF.

  ENDIF.


*  LOOP AT lt_vbrk INTO DATA(ls_vbrk).
*
*    READ TABLE  lt_vbrp INTO DATA(ls_vbrp) WITH KEY vbeln = ls_vbrk-vbeln.
*    IF sy-subrc = 0.
*
*      READ TABLE gt_zwm_invc_put INTO DATA(ls_invc_put) WITH KEY delivery_no = ls_vbrp-vgbel.
*      IF sy-subrc = 0.
*
*        READ TABLE lt_vbuk INTO DATA(ls_vbuk) WITH KEY vbeln = ls_vbrp-vgbel.
*        IF sy-subrc = 0.
*
*          ls_final-chk_box = 'X'.
*          ls_final-lgnum = ls_invc_put-lgnum.
*          ls_final-likp_vbeln  = ls_invc_put-delivery_no.
*          ls_final-vbeln = ls_vbrp-vbeln.
*          ls_final-erdat = ls_vbrk-erdat.
*          ls_final-kunrg = ls_vbrk-kunrg.

*          READ TABLE lt_kna1 INTO DATA(ls_kna1) WITH KEY kunnr = ls_vbrk-kunrg.
*          IF sy-subrc = 0.
*            ls_final-name1 = ls_kna1-name1.
*          ENDIF.
*
*
*          APPEND ls_final TO gt_final.
*          CLEAR : ls_final.
*
*        ENDIF.
*      ENDIF.
*    ENDIF.
*
*    CLEAR :ls_vbrk.
*  ENDLOOP.


  LOOP AT  gt_zwm_invc_put INTO DATA(ls_invc_put).

    ls_final-chk_box = 'X'.
    ls_final-lgnum = ls_invc_put-lgnum.
    ls_final-likp_vbeln  = ls_invc_put-delivery_no.

    READ TABLE lt_vbrp INTO DATA(ls_vbrp) WITH KEY vgbel = ls_invc_put-delivery_no
                                                   vgpos = ls_invc_put-posnr.

    IF sy-subrc = 0.
      ls_final-vbeln = ls_vbrp-vbeln.

      READ TABLE lt_vbrk INTO DATA(ls_vbrk) WITH KEY vbeln = ls_vbrp-vbeln.
      IF sy-subrc = 0.
        ls_final-erdat = ls_vbrk-erdat.
        ls_final-kunrg = ls_vbrk-kunrg.

        READ TABLE lt_kna1 INTO DATA(ls_kna1) WITH KEY kunnr = ls_vbrk-kunrg.
        IF sy-subrc = 0.
          ls_final-name1 = ls_kna1-name1.

          CALL FUNCTION 'CONVERSION_EXIT_ALPHA_OUTPUT'
            EXPORTING
              input  = ls_final-kunrg
            IMPORTING
              output = ls_final-kunrg.
        ENDIF.
      ENDIF.
      APPEND ls_final TO gt_final.
      CLEAR : ls_final.
    ENDIF.

    CLEAR: ls_invc_put,ls_final.
  ENDLOOP.

  DELETE ADJACENT DUPLICATES FROM gt_final COMPARING ALL FIELDS .


*    LOOP AT lt_vbrk INTO ls_vbrk.

  ls_fieldcat-seltext_m = 'WareHouse No'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.

  CLEAR : ls_fieldcat.
  lv_index = lv_index + 1.
  ls_fieldcat-fieldname = 'VBELN'.
  ls_fieldcat-seltext_m = 'Invoice Number'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.

  CLEAR : ls_fieldcat.
  lv_index = lv_index + 1.
  ls_fieldcat-fieldname = 'LIKP_VBELN'.
  ls_fieldcat-seltext_m = 'Delivery Number'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.

  CLEAR : ls_fieldcat.
  lv_index = lv_index + 1.
  ls_fieldcat-fieldname = 'ERDAT'.
  ls_fieldcat-seltext_m = 'Invoice Date'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.

  CLEAR : ls_fieldcat.
  lv_index = lv_index + 1.
  ls_fieldcat-fieldname = 'KUNRG'.
  ls_fieldcat-seltext_m = 'Payer'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.

  CLEAR : ls_fieldcat.
  lv_index = lv_index + 1.
  ls_fieldcat-fieldname = 'NAME1'.
  ls_fieldcat-seltext_m = 'Payer Name'.
  ls_fieldcat-col_pos = lv_index.
  APPEND ls_fieldcat TO gt_fieldcat.



  PERFORM display_alv.



FORM status USING pfstat TYPE slis_t_extab.
  SET PF-STATUS 'ZWM_INVC'.
ENDFORM.

FORM usercommand USING r_ucomm     LIKE sy-ucomm
                       rs_selfield TYPE slis_selfield.

  FIELD-SYMBOLS: <fs_final> TYPE zwm_invc_put ."ty_final.
  rs_selfield-exit = 'X'.

  DATA: ref_grid TYPE REF TO cl_gui_alv_grid.
  IF ref_grid IS INITIAL.
    CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
      IMPORTING
        e_grid = ref_grid.
  ENDIF.

  IF NOT ref_grid IS INITIAL.
    CALL METHOD ref_grid->check_changed_data.
  ENDIF.


  CASE r_ucomm.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE TO SCREEN 0.
    WHEN 'CANCEL'.
      LEAVE TO SCREEN 0.
    WHEN 'SALL'.
*      BREAK-POINT.
      LOOP AT gt_final ASSIGNING FIELD-SYMBOL(<fs_final1>).
        <fs_final1>-chk_box = 'X'.
      ENDLOOP.

      PERFORM display_alv.

    WHEN 'DALL'.
*      BREAK-POINT.
      LOOP AT gt_final ASSIGNING <fs_final1>.
        CLEAR <fs_final1>-chk_box .
      ENDLOOP.

      PERFORM display_alv.

    WHEN 'CLEAR'.
      PERFORM save.
    WHEN OTHERS.
  ENDCASE.

ENDFORM.
