form table_selection .

  if p_bkpf = 'X'.

    perform input_validation.
    if gv_flag ne 'X'.
      perform get_bkpf.
    endif.
    exit.
  elseif p_bseg = 'X'.
    perform input_validation.
    if gv_flag ne 'X'.
      perform get_bkpf1.
      perform get_bseg.
    endif.
  elseif p_skat = 'X'.
    perform get_skat.
  elseif p_glt0 = 'X'.
    perform get_glt0.
  endif.

endform.

*&---------------------------------------------------------------------*
form input_validation .
  clear gv_flag.
  if p_datef = '' or p_datef = '00000000'.
    message 'From date can not be blank' type 'I'.
    gv_flag = 'X'.
    exit.
  elseif p_datet  = '' or p_datet = '00000000'.
    message 'To date can not be blank' type 'I'.
    gv_flag = 'X'.
    exit.
  elseif p_bukrs is initial.
    message 'Company Code can not be blank' type 'I'.
    gv_flag = 'X'.
    exit.
  endif.

endform.

form get_skat .

  select
         ktopl
         saknr
         txt20
         txt50
    from skat
    into table it_skat
    where spras = 'EN'.

  if it_skat is not initial.

    clear f_path.
    data: dsn     type string,
          rec(80).
    data : v_structure type string.

    dsn = p_flname.
    translate dsn to lower case.

    open dataset dsn for output in text mode encoding default.

    read table it_skat into wa_skat index 1.
    if sy-subrc eq 0.
      concatenate 'KTOPL'
                   'SAKNR'
                   'TXT20'
                   'TXT50'
                    into v_structure
                        separated by cr_lf  ."SEPARATED BY cl_abap_char_utilities=>horizontal_tab.
      transfer v_structure to dsn.
      clear wa_skat.
    endif.

    read table it_skat into wa_skat index 1.
    if sy-subrc eq 0.
      concatenate 'Chrt/Accts'
                  'G/L Account'
                  'Short Text'
                  'Long Text'
                  into v_structure
                        separated by cr_lf  ."SEPARATED BY cl_abap_char_utilities=>horizontal_tab.
      transfer v_structure to dsn.
      clear wa_skat.
    endif.

    loop at it_skat into wa_skat.

      concatenate wa_skat-ktopl
                  wa_skat-saknr
                  wa_skat-txt20
                  wa_skat-txt50
                  into v_structure
                    separated by cr_lf  ."SEPARATED BY cl_abap_char_utilities=>horizontal_tab.
      transfer v_structure to dsn.
      clear wa_skat.
    endloop.
    close dataset dsn.

    clear f_path.

  endif.
endform.
