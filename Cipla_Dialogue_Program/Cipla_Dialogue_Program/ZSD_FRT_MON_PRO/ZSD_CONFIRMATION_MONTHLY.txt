*----------------------------------------------------------------------*
FORM fetch_data .

  DATA: wa_date     TYPE d,
        s_date      LIKE sy-datum,
        e_date      LIKE sy-datum,
        l_date      TYPE d,
        l_month1(2) TYPE c,
        l_year1(4)  TYPE c.


*&---------------------------------------------------------------------*
*& Previous Month Date Calculate
*&---------------------------------------------------------------------*

  SELECT * FROM zsd_frt_plt_dt INTO TABLE ta_zsd_frt_plt_dt
    WHERE werks = l_werks.

  SORT ta_zsd_frt_plt_dt DESCENDING BY werks month_d gjahr.

  CONCATENATE l_year l_month co_01 INTO wa_date .


  CALL FUNCTION 'OIL_LAST_DAY_OF_PREVIOUS_MONTH'
    EXPORTING
      i_date_old = wa_date
    IMPORTING
      e_date_new = l_date.


  l_month1 = l_date+4(2).
  l_year1  = l_date+0(4).

  READ TABLE ta_zsd_frt_plt_dt INTO wa_zsd_frt_plt_dt WITH KEY month_d = l_month1 gjahr = l_year1.

  IF sy-subrc  = 0.
    wa_zsd_frt_plt_dt-werks   = l_werks.
    wa_zsd_frt_plt_dt-month_d = l_month.
    wa_zsd_frt_plt_dt-gjahr   = l_year.

    MODIFY zsd_frt_plt_dt FROM wa_zsd_frt_plt_dt.
    CLEAR wa_zsd_frt_plt_dt.
  ELSE.
    MESSAGE co_msg1 TYPE co_e.
  ENDIF.

*&---------------------------------------------------------------------*
*& Previous Month First & Last Date Calculate
*&---------------------------------------------------------------------*

  CALL FUNCTION 'HR_JP_MONTH_BEGIN_END_DATE'
    EXPORTING
      iv_date             = wa_date
    IMPORTING
      ev_month_begin_date = s_date
      ev_month_end_date   = e_date.

*&---------------------------------------------------------------------*
*& Fetch data from table
*&---------------------------------------------------------------------*
  SELECT werks zconpl zcnf FROM zsd_combin_plant
    INTO TABLE ta_plant
    WHERE werks = l_werks.

  SORT ta_plant BY werks .
  DELETE ADJACENT DUPLICATES FROM ta_plant COMPARING werks.

* delete ta_plant where zcnf = 'X'.


  SELECT * FROM zsd_frt_inout_tr INTO TABLE ta_inout " For Fetch data from Trip Id table
    FOR ALL ENTRIES IN ta_plant
      WHERE werks = ta_plant-werks
        AND zcan = co_x
       AND zdispdt GE s_date
       AND zdispdt LE e_date
       AND ztraty = co_mon.

  DELETE ta_inout WHERE zprov_trfr_flag = co_x.


  IF ta_inout[] IS NOT INITIAL.
    SELECT *
     FROM zsd_one_time_tri
     INTO TABLE ta_one
      FOR ALL ENTRIES IN ta_inout
     WHERE werks = ta_inout-werks
        AND zfreight_type  = ta_inout-zfreight_type
        AND lifnr  = ta_inout-ztranc
        AND ztrip_id  = ta_inout-ztripid
        AND zrate_type  = ta_inout-ztraty
        AND zdispdt      = ta_inout-zdispdt .

    SORT ta_one BY werks zfreight_type lifnr .

    SELECT mandt
           zlr_type
           zfreight_on
           zfreight_type
           kunnr
           kunag
           ztripid
           zlrsmo
           zinvice
           bukrs
           gjahr
           vkorg
           vtweg
           spart
           werks
           zconpl
           zmtripid
           zmod_disp_code
