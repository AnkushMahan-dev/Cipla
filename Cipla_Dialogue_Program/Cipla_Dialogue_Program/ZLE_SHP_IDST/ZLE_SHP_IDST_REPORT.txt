*&---------------------------------------------------------------------*
*&      Form  FETCH_F4HELP_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM fetch_f4help_data .

  REFRESH : gt_minv,
            gt_free.

  SELECT docno INTO TABLE gt_minv
    FROM zle_mms_free
   WHERE blart = co_mi."'MI'.
  IF sy-subrc  = 0.
    SORT gt_minv BY docno.
  ENDIF.


  SELECT docno INTO TABLE gt_free
    FROM zle_mms_free
   WHERE blart IN (co_fs, co_ot)."('FS','OT').
  IF sy-subrc  = 0.
    SORT gt_free BY docno.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  VALIDATE_TPLST
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM validate_tplst .

  TYPES : BEGIN OF ty_ttds,
            tplst TYPE tplst,
          END OF ty_ttds.

  DATA  : lt_ttds TYPE STANDARD TABLE OF ty_ttds,
          ls_ttds TYPE ty_ttds.

  SELECT tplst INTO TABLE lt_ttds
    FROM ttds
   WHERE tplst IN s_tplst.
  IF sy-subrc = 0.
    gv_message = co_youare."'You are not authorized for the tr. pl. pt. '' '.
    LOOP AT lt_ttds INTO ls_ttds.

      AUTHORITY-CHECK OBJECT 'V_VTTK_TDS'
        ID 'TPLST' FIELD ls_ttds-tplst.
      IF sy-subrc <> 0.
        gv_lineno = gv_lineno + 1.
        CLEAR gs_messages.
        gs_messages-msgid   = co_zle_shp_msg."'ZLE_SHP_MSG'.
        gs_messages-msgty   = co_e."'E'.
        gs_messages-msgno   = 000.
        gs_messages-msgv1   = gv_message.
        gs_messages-msgv2   = ls_ttds-tplst.
        gs_messages-msgv3   = ''''.
        gs_messages-lineno  = gv_lineno.
        APPEND gs_messages TO gt_messages.
      ENDIF.
    ENDLOOP.

    IF gt_messages IS NOT INITIAL.
      PERFORM display_messages.
    ENDIF.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  DISPLAY_INPUTS
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_GFIELD  text
*----------------------------------------------------------------------*
FORM display_inputs  USING    p_field.

  DATA  : lv_field TYPE fieldname,
          lv_title TYPE c LENGTH 20.

  FIELD-SYMBOLS : <ft_f4table> TYPE STANDARD TABLE.

  IF p_field CS co_mmsinv."'MMSINV'.
    lv_field  = co_docno."'DOCNO'.
    lv_title  = co_mmsi."'MMS Invoices'.
    ASSIGN gt_minv TO <ft_f4table>.
  ELSEIF p_field CS co_free."'FREE'.
    lv_field  = co_docno."'DOCNO'.
    lv_title  = co_frees."'Free Samples/Others'.
    ASSIGN gt_free TO <ft_f4table>.
  ENDIF.

  IF <ft_f4table> IS ASSIGNED.

    REFRESH: gt_return.

    CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
      EXPORTING
        retfield        = lv_field
        dynpprog        = sy-cprog
        dynpnr          = sy-dynnr
        dynprofield     = p_field
        window_title    = lv_title
        value_org       = 'S'
