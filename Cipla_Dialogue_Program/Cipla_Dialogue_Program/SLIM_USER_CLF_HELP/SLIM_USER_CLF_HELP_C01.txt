" Replacement for CL_LAW3_USER_TYPES
CLASS lcl_user_types DEFINITION.

  PUBLIC SECTION.
    TYPES: ty_law2_usertypes TYPE char2.
    TYPES: tyt_law2_usertypes TYPE STANDARD TABLE OF ty_law2_usertypes.

    TYPES: BEGIN OF tys_usertypes_resolved,
             usertype TYPE ty_law2_usertypes,
             resolved TYPE char1,
           END OF tys_usertypes_resolved.
    TYPES: tyt_usertypes_resolved TYPE SORTED TABLE OF tys_usertypes_resolved WITH NON-UNIQUE KEY usertype.

    TYPES: BEGIN OF tys_law_cont_included,
             usertype    TYPE ty_law2_usertypes,
             included_in TYPE tyt_usertypes_resolved,
           END OF tys_law_cont_included.
    TYPES: tyt_law_cont_included TYPE TABLE OF tys_law_cont_included.

    TYPES: tyt_law_cont TYPE STANDARD TABLE OF law_cont.

    DATA: gt_law_cont_resolution TYPE tyt_law_cont_included.

    METHODS:
      consolidate_user_types
        IMPORTING it_user_types  TYPE tyt_law2_usertypes
        EXPORTING ev_combination TYPE char100.

  PRIVATE SECTION.

    METHODS:
      determine_law_cont_resolution
        EXPORTING et_law_cont_res TYPE tyt_law_cont_included,

      get_law_cont
        EXPORTING et_law_cont TYPE tyt_law_cont,

      fill_user_type_combinations
        IMPORTING it_type                TYPE tyt_law2_usertypes
                  it_law_cont_resolution TYPE tyt_law_cont_included
        EXPORTING et_type                TYPE tyt_law2_usertypes
                  ev_combination         TYPE char100
                  ev_multiple_types      TYPE flag.

ENDCLASS.

CLASS lcl_user_types IMPLEMENTATION.

  METHOD consolidate_user_types.

    CLEAR ev_combination.

    IF gt_law_cont_resolution IS INITIAL.
      " Fill the global table
      me->determine_law_cont_resolution(
        IMPORTING
          et_law_cont_res = gt_law_cont_resolution ).
    ENDIF.

    me->fill_user_type_combinations(
      EXPORTING
        it_type                = it_user_types
        it_law_cont_resolution = gt_law_cont_resolution
      IMPORTING
       ev_combination          = ev_combination ).

  ENDMETHOD.

  METHOD fill_user_type_combinations.

    TYPES: BEGIN OF tys_unique_user_usertype,
             uniquelawguid TYPE  guid,
             uname         TYPE  xubname,
             licensetype   TYPE  ty_law2_usertypes,
           END OF tys_unique_user_usertype.

    TYPES: BEGIN OF tys_law2_usercons,
             datasetguid     TYPE guid,
             usertype        TYPE ty_law2_usertypes,
             usercombination TYPE char100,
             counter         TYPE int4,
           END OF tys_law2_usercons.

    DATA:
      lv_prev_guid    TYPE guid,
      lt_type         TYPE tyt_law2_usertypes,
      ls_combination  TYPE tys_law2_usercons,
      lv_result_type  TYPE ty_law2_usertypes,
      lv_target_value TYPE i,
      lt_type_tmp     TYPE tyt_law2_usertypes,
      lv_tabix        LIKE sy-tabix,
      lv_is_value     TYPE i.

    FIELD-SYMBOLS: <fs_type>     TYPE ty_law2_usertypes,
                   <fs_type2>    TYPE ty_law2_usertypes,
                   <fs_law_cont> TYPE tys_law_cont_included,
                   <fs_usertype> TYPE tys_unique_user_usertype.

    CLEAR ev_combination.

    lt_type[] = it_type[].
    SORT lt_type ASCENDING.
    " Check for each type if it includes all other types:
    " Determine the expected counter for success:
    lv_target_value = lines( lt_type ) - 1.
    LOOP AT lt_type ASSIGNING <fs_type>.
      CLEAR: lv_is_value, lv_result_type.
      " Get the LAW_CONT entry:
      READ TABLE it_law_cont_resolution ASSIGNING <fs_law_cont> WITH KEY usertype = <fs_type> BINARY SEARCH.
      IF sy-subrc NE 0.
        " Current user type does not include any other type ...
        CONTINUE.
      ENDIF.

      " Check for all types if they are in the current 'included_in' table:
      LOOP AT lt_type ASSIGNING <fs_type2>.
        CHECK <fs_type> NE <fs_type2>.
        READ TABLE <fs_law_cont>-included_in WITH KEY usertype = <fs_type2> TRANSPORTING NO FIELDS BINARY SEARCH.
        IF sy-subrc NE 0.
          " Not included -> exit the loop.
          EXIT.
        ELSE.
          " Included
          ADD 1 TO lv_is_value.
        ENDIF.
      ENDLOOP.

      " Current type includes all other types:
      IF lv_is_value = lv_target_value.
        ev_combination = <fs_type>.
        EXIT.
      ENDIF.
    ENDLOOP.

    IF NOT ev_combination IS INITIAL.
      RETURN.
    ENDIF.

    "-------------------------------------------
    " Create the result combination:
    "-------------------------------------------
    CLEAR ls_combination.

    ls_combination-counter = 1.
    " Result for the current combination
    IF lv_result_type IS NOT INITIAL.
      ls_combination-usertype = lv_result_type.
    ELSE.
      " Try to do a partly resolution
      lt_type_tmp[] = lt_type[].
      LOOP AT lt_type ASSIGNING <fs_type>.
        lv_tabix = sy-tabix.
        " Check if any of the given user types is included in another type from table lt_type:
        LOOP AT lt_type ASSIGNING <fs_type2>.
          CHECK <fs_type> NE <fs_type2>.
          READ TABLE it_law_cont_resolution ASSIGNING <fs_law_cont> WITH KEY usertype = <fs_type2> BINARY SEARCH.
          IF sy-subrc NE 0.
            " Current user type does not include any other type ...
            CONTINUE.
          ENDIF.
          READ TABLE <fs_law_cont>-included_in WITH KEY usertype = <fs_type> TRANSPORTING NO FIELDS BINARY SEARCH.
          IF sy-subrc EQ 0.
            " <fs_type> is included in another type -> delete it from the table lt_type.
            DELETE lt_type INDEX lv_tabix.
            EXIT.
          ENDIF.
        ENDLOOP.
      ENDLOOP.

    ENDIF.

    " Concatenate the cobinations
    LOOP AT lt_type ASSIGNING <fs_type>.
      IF sy-tabix = 1.
        ls_combination-usercombination = <fs_type>.
      ELSE.
        CONCATENATE ls_combination-usercombination '#' <fs_type> INTO ls_combination-usercombination.
      ENDIF.
    ENDLOOP.

    ev_combination = ls_combination-usercombination.

  ENDMETHOD.

  METHOD get_law_cont.

    DATA: ls_law_cont TYPE law_cont,
          lv_tabname  TYPE string VALUE 'LAW2D_CONT'.

    IF gt_law_cont_resolution IS INITIAL.

      " Standard Consolidation Rules
      SELECT * FROM law_cont INTO TABLE et_law_cont.

      " Check whether LAW2D_CONT exists in the current system
      CALL METHOD cl_abap_structdescr=>describe_by_name
        EXPORTING
          p_name = lv_tabname
        EXCEPTIONS
          OTHERS = 1.

      IF sy-subrc = 0.
        " Customer-specific Consolidation Rules (via LAW Upload)
        SELECT usertyp containsu FROM (lv_tabname) INTO CORRESPONDING FIELDS OF ls_law_cont.
          INSERT ls_law_cont INTO TABLE et_law_cont.
        ENDSELECT.
      ENDIF.

      " Delete duplicates
      SORT et_law_cont ASCENDING.
      DELETE ADJACENT DUPLICATES FROM et_law_cont COMPARING usertyp containsu.

    ENDIF.

  ENDMETHOD.

  METHOD determine_law_cont_resolution.

    DATA: lt_law_cont      TYPE TABLE OF law_cont,
          lv_prev_type     TYPE ty_law2_usertypes,
          ls_included_in   TYPE tys_usertypes_resolved,
          ls_resolved_type TYPE tys_law_cont_included,
          lt_resolved_type TYPE tyt_law_cont_included,
          lv_change        TYPE i.

    FIELD-SYMBOLS: <fs_law_cont>       TYPE law_cont,
                   <fs_resolve_type>   TYPE tys_law_cont_included,
                   <fs_2_resolve_type> TYPE tys_law_cont_included,
                   <fs_2_included>     TYPE tys_usertypes_resolved,
                   <fs_included_type>  TYPE tys_usertypes_resolved.

    CLEAR et_law_cont_res.

    me->get_law_cont( IMPORTING et_law_cont = lt_law_cont ).

    "-----------------------------------------------------
    " Build the first level include list
    "-----------------------------------------------------
    SORT lt_law_cont BY usertyp ASCENDING.
    LOOP AT lt_law_cont ASSIGNING <fs_law_cont>.
      CLEAR ls_included_in.
      " First entry:
      IF sy-tabix = 1.
        lv_prev_type              = <fs_law_cont>-usertyp.
        ls_resolved_type-usertype = <fs_law_cont>-usertyp.
      ENDIF.
      " Same usertype?
      IF lv_prev_type = <fs_law_cont>-usertyp.
        ls_included_in-usertype = <fs_law_cont>-containsu.
        INSERT ls_included_in INTO TABLE ls_resolved_type-included_in.
        " Next usertype
      ELSE.
        " Append the previous entry
        APPEND ls_resolved_type TO lt_resolved_type.
        " Create a new entry
        CLEAR ls_resolved_type.
        ls_resolved_type-usertype = <fs_law_cont>-usertyp.
        " Insert the included user type
        ls_included_in-usertype   = <fs_law_cont>-containsu.
        INSERT ls_included_in INTO TABLE ls_resolved_type-included_in.
        lv_prev_type              = <fs_law_cont>-usertyp.
      ENDIF.
    ENDLOOP.

    " Append the last usertype entry
    APPEND ls_resolved_type TO lt_resolved_type.
    CLEAR ls_resolved_type.

    " Resolve the next level inheritance:
    " (Loop until all included user types are resolved)
    DO.
      " Indicator if a change was done in the current loop
      CLEAR lv_change.

      LOOP AT lt_resolved_type ASSIGNING <fs_resolve_type>.
        LOOP AT <fs_resolve_type>-included_in ASSIGNING <fs_included_type>.
          " Check if the included type was already resolved -> nothing to do for this entry:
          CHECK <fs_included_type>-resolved IS INITIAL.
          " Get the included user types of the current 'include' type:
          READ TABLE lt_resolved_type ASSIGNING <fs_2_resolve_type> WITH KEY usertype = <fs_included_type>-usertype BINARY SEARCH.
          " If there are entries which are not included in the 'to be resolved' user type insert it into the included table:
          IF sy-subrc = 0.
            LOOP AT <fs_2_resolve_type>-included_in ASSIGNING <fs_2_included>.
              READ TABLE <fs_resolve_type>-included_in WITH KEY usertype = <fs_2_included>-usertype TRANSPORTING NO FIELDS BINARY SEARCH.
              IF sy-subrc <> 0.
                INSERT <fs_2_included> INTO TABLE <fs_resolve_type>-included_in.
                ADD 1 TO lv_change.
              ENDIF.
            ENDLOOP.
          ENDIF.
          " Current entry is resolved
          <fs_included_type>-resolved = abap_true.
        ENDLOOP.
      ENDLOOP.

      " If no change was done within the last loop, we can exit the loop processing
      IF lv_change = 0.
        EXIT.
      ENDIF.
    ENDDO.

    " Return the result
    et_law_cont_res = lt_resolved_type[].

  ENDMETHOD.

ENDCLASS.

*---------------------------------------------------------------------*
*       CLASS lcl_handle_events IMPLEMENTATION
*---------------------------------------------------------------------*
CLASS lcl_handle_events IMPLEMENTATION.

  METHOD on_single_click_alv1. " Selection on Result Screen 0400
    PERFORM show_classif_totals  USING row column.
  ENDMETHOD.

  METHOD on_single_click_alv2.

    CASE column.
      WHEN 'AGR_NAME' OR 'AUTH'.
        PERFORM show_classif_objects_user USING row.
      WHEN 'TOTAL_USR'.
        PERFORM show_classif_objects_role USING row column.
    ENDCASE.

  ENDMETHOD.

  METHOD on_double_click_0403.
    PERFORM show_transactions USING row.
  ENDMETHOD.

  METHOD on_single_click_alv3.
    PERFORM show_classif_objects_role  USING row column.
  ENDMETHOD.

  METHOD on_single_click_alv4. " User Selection on Screen 0401
    PERFORM show_classif_roles  USING abap_false row.
  ENDMETHOD.

  METHOD on_user_command.

    DATA: lv_row     TYPE salv_de_row,
          lv_error   TYPE boolean.

    CASE e_salv_function.

      WHEN '&DETAILS'.
        PERFORM download_details.

      WHEN '&TOGGLE'.
        " Show/Hide unclassified objects
        CASE sy-dynnr.
          WHEN 403.
            " Show / Hide unclassified entries
            PERFORM show_hide_unclassified_entries USING sy-dynnr CHANGING gv_403_filtered.
            " Show the updated list
            PERFORM pbo_0403.

          WHEN 402.
            " Show / Hide unclassified entries
            PERFORM show_hide_unclassified_entries USING sy-dynnr CHANGING gv_402_filtered.
            " Show the updated list
            PERFORM pbo_0402.

          WHEN 500.
            " Show / Hide unclassified entries
            PERFORM show_hide_unclassified_entries USING sy-dynnr CHANGING gv_500_filtered.
            " Show the updated list
            PERFORM pbo_0500.

        ENDCASE.

      WHEN '&USERS'.
      " Simulate Users
      PERFORM simulate_users_role_validation.

      WHEN '&EXP_UV' OR '&EXP_SAP'.
      " Export result of User Validation
      IF e_salv_function = '&EXP_UV'.
        gv_export_type = gc_export_type-user.
      ELSEIF e_salv_function = '&EXP_SAP'.
        gv_export_type = gc_export_type-sap.
      ENDIF.
      PERFORM password_request CHANGING lv_error.
      IF lv_error EQ abap_true.
        RETURN.
      ENDIF.
      PERFORM export_validation USING gc_valid_type-user_validation.

      WHEN '&EXP_RV' OR '&EXP_RV_SAP'.
      " Export result of Role Validation
      IF e_salv_function = '&EXP_RV'.
        gv_export_type = gc_export_type-user.
      ELSEIF e_salv_function = '&EXP_RV_SAP'.
        gv_export_type = gc_export_type-sap.
      ENDIF.
      PERFORM password_request CHANGING lv_error.
      IF lv_error EQ abap_true.
        RETURN.
      ENDIF.
      PERFORM export_validation USING gc_valid_type-role_validation.

      WHEN '&ROLES'.
        " Role display for selected users
        PERFORM show_classif_roles USING abap_true lv_row.

      WHEN '&ROLECLASS'.
        " Transfer role classifications Tx license_attributes
        PERFORM transfer_role_classifications.

      WHEN '&USERCLASS'.
        " Transfer the user classifications to USR06
        PERFORM transfer_user_classifications.

    ENDCASE.

  ENDMETHOD.

ENDCLASS.                    "lcl_handle_events IMPLEMENTATION
