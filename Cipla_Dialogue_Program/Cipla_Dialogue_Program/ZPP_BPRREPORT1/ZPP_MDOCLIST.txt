  ENDIF   .

    WHEN 4.

      CONCATENATE l_exptyp  ' and ' l_section INTO l_section SEPARATED BY space.
      PERFORM get_main_data USING l_section .
    WHEN 5.
      CONCATENATE   'MARKT_AUTH_NO EQ ''' 'X' '''' INTO l_section .
      CONCATENATE l_exptyp  ' and ' l_section INTO l_section SEPARATED BY space.
      PERFORM get_main_data USING l_section .
    WHEN 6.
      CONCATENATE  'CUST_BATCH EQ ''' 'X' '''' INTO l_section .
