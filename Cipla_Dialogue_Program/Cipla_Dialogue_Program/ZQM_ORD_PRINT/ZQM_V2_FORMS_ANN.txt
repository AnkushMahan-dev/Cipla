FORM dynamic_heading .
  CALL METHOD OF g_selection 'Expand' EXPORTING #1 = 6.
  CALL METHOD OF g_selection 'EndKey' EXPORTING #1 = 5.
  CALL METHOD OF g_selection 'MoveLeft'.
  CALL METHOD OF g_selection 'InsertRows' EXPORTING #1 = 1.
  CALL METHOD OF g_selection 'Collapse'.
  CALL METHOD OF g_selection 'MoveRight' EXPORTING #1 = 1 #2 = 6 #3 = 1.
  GET PROPERTY OF g_selection 'Cells' = g_cell.
  CALL METHOD OF g_cell 'Merge'.
  CALL METHOD OF g_selection 'MoveLeft'.
  PERFORM set_font.
  PERFORM set_bold USING 1.
  IF doc_type = 'ZAN'.
    PERFORM set_text_alignment USING 1.
    PERFORM write_text USING c_hd_heading_bmr.
  ELSEIF doc_type = 'BPR'.
    PERFORM set_text_alignment USING 0.
    PERFORM write_text USING c_hd_heading_bpr.
  ENDIF.
*  PERFORM new_line.
ENDFORM.                    " DYNAMIC_HEADING

FORM get_footer_image .
  DATA : l_bytecount TYPE i,
         l_tdbtype   LIKE stxbitmaps-tdbtype,
         l_content   TYPE STANDARD TABLE OF bapiconten INITIAL SIZE 0.

  DATA: graphic_size TYPE i,
        wa_bmr_logo  TYPE zpp_bmr_logo,
        wa_ann_logo  TYPE zpp_ann_logo.


  DATA: BEGIN OF graphic_table OCCURS 0,
          line(255) TYPE x,
        END OF graphic_table.

  DATA : ta_dd07v TYPE TABLE OF dd07v,
         wa_dd07v TYPE dd07v.

  SELECT SINGLE * FROM zpp_form_plant INTO @DATA(wa_zpp_form_plant)
    WHERE werks_ass = @p_plant.
*COmmented by Gajanan - 07.07.2025 - CR#8000003007

*    EXPORTING
*      domname         = 'Z_PLT_TYP'
*      text            = 'X'
*    TABLES
*      values_tab      = ta_dd07v
*    EXCEPTIONS
*      no_values_found = 1
*      OTHERS          = 2.
*  IF sy-subrc EQ 0.
*    CLEAR wa_dd07v.
*    READ TABLE ta_dd07v INTO wa_dd07v WITH KEY domvalue_l = wa_zpp_form_plant-z_plt_typ.
*    IF sy-subrc = 0.
*      SELECT SINGLE * FROM  zpp_bmr_logo CLIENT SPECIFIED
*        INTO wa_bmr_logo
*             WHERE  mandt   = sy-mandt
*             AND    bmrtyp  = wa_dd07v-ddtext.
*    ENDIF.
*  ENDIF.
  IF sy-subrc = 0.
    SELECT SINGLE * FROM  zpp_ann_logo CLIENT SPECIFIED
      INTO wa_ann_logo
           WHERE  mandt   = sy-mandt
             AND anntyp = 'ANN'
             AND z_plt_typ  = wa_zpp_form_plant-z_plt_typ.
  ENDIF.

  CALL FUNCTION 'SAPSCRIPT_GET_GRAPHIC_BDS'
    EXPORTING
      i_object       = 'GRAPHICS'
      i_name         = wa_ann_logo-logo_nm
      i_id           = 'BMAP'
      i_btype        = 'BMON'
    IMPORTING
      e_bytecount    = l_bytecount
    TABLES
      content        = l_content
    EXCEPTIONS
      not_found      = 1
      bds_get_failed = 2
      bds_no_content = 3
      OTHERS         = 4.

  CALL FUNCTION 'SAPSCRIPT_CONVERT_BITMAP'
    EXPORTING
      old_format               = 'BDS'
      new_format               = 'BMP'
      bitmap_file_bytecount_in = l_bytecount
    IMPORTING
      bitmap_file_bytecount    = graphic_size
    TABLES
      bds_bitmap_file          = l_content
      bitmap_file              = graphic_table
    EXCEPTIONS
      OTHERS                   = 1.


  CONCATENATE 'C:\temp\' doc_type wa_ann_logo-logo_nm '.bmp' INTO g_logo_path.
  CALL FUNCTION 'WS_DOWNLOAD'
    EXPORTING
      bin_filesize            = graphic_size
      filename                = g_logo_path      "'C:\temp\123.bmp'
      filetype                = 'BIN'
    TABLES
      data_tab                = graphic_table
    EXCEPTIONS
      invalid_filesize        = 1
      invalid_table_width     = 2
      invalid_type            = 3
      no_batch                = 4
      unknown_error           = 5
      gui_refuse_filetransfer = 6.

  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  ENDIF.
ENDFORM.                    " GET_FOOTER_IMAGE
